424  public static <T extends Enum<T>> Enum<T> findEnum(String name,
425  Class<? extends Enum<T>> enumClass) {
426 
427  if (name == null)
428  return null;
429  name = name.trim();
430  for (Enum<T> e : enumClass.getEnumConstants()) {
431  if (e.name().toLowerCase(Locale.ENGLISH).replace("_", "").equals(name)) {
432  return e;
433  }
434  }
435  return null;
436  }

448  public static double stringToDouble(String s) throws NumberFormatException {
449  if (s == null)
450  throw new NumberFormatException("null string");
451  if (s.equalsIgnoreCase("NaN"))
452  return Double.NaN;
453  if (s.equalsIgnoreCase("Inf"))
