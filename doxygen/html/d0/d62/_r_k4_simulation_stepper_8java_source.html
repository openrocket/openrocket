128  * dt[1]: the value of maxTimeStep

316  private void calculateAcceleration(RK4SimulationStatus status, DataStore store) throws SimulationException {
317 
318  // Call pre-listeners
319  store.accelerationData = SimulationListenerHelper.firePreAccelerationCalculation(status);
320  if (store.accelerationData != null) {
321  return;
322  }
323 
324  // Compute the forces affecting the rocket
325  calculateForces(status, store);
326 
327  // Calculate mass data
328  store.massData = calculateMassData(status);
329 
330 
331  // Calculate the forces from the aerodynamic coefficients
332 
333  double dynP = (0.5 * store.flightConditions.getAtmosphericConditions().getDensity() *
334  MathUtil.pow2(store.flightConditions.getVelocity()));
335  double refArea = store.flightConditions.getRefArea();
336  double refLength = store.flightConditions.getRefLength();
337 
338 
339  // Linear forces in rocket coordinates
340  store.dragForce = store.forces.getCaxial() * dynP * refArea;
341  double fN = store.forces.getCN() * dynP * refArea;
342  double fSide = store.forces.getCside() * dynP * refArea;
343 
344  double forceZ = store.thrustForce - store.dragForce;
345 
346  store.linearAcceleration = new Coordinate(-fN / store.massData.getCG().weight,
347  -fSide / store.massData.getCG().weight,
348  forceZ / store.massData.getCG().weight);
349 
350  store.linearAcceleration = store.thetaRotation.rotateZ(store.linearAcceleration);
351 
352  // Convert into rocket world coordinates
353  store.linearAcceleration = status.getRocketOrientationQuaternion().rotate(store.linearAcceleration);
354 
355  // add effect of gravity
356  store.gravity = modelGravity(status);
357  store.linearAcceleration = store.linearAcceleration.sub(0, 0, store.gravity);
358 
359  // add effect of Coriolis acceleration
360  store.coriolisAcceleration = status.getSimulationConditions().getGeodeticComputation()
361  .getCoriolisAcceleration(status.getRocketWorldPosition(), status.getRocketVelocity());
362  store.linearAcceleration = store.linearAcceleration.add(store.coriolisAcceleration);
363 
364  // If still on the launch rod, project acceleration onto launch rod direction and
365  // set angular acceleration to zero.
366  if (!status.isLaunchRodCleared()) {
367 
368  store.linearAcceleration = status.getLaunchRodDirection().multiply(
369  store.linearAcceleration.dot(status.getLaunchRodDirection()));
370  store.angularAcceleration = Coordinate.NUL;
371  store.rollAcceleration = 0;
372  store.lateralPitchAcceleration = 0;
373 
374  } else {
375 
376  // Shift moments to CG
377  double Cm = store.forces.getCm() - store.forces.getCN() * store.massData.getCG().x / refLength;
378  double Cyaw = store.forces.getCyaw() - store.forces.getCside() * store.massData.getCG().x / refLength;
379 
380  // Compute moments
381  double momX = -Cyaw * dynP * refArea * refLength;
382  double momY = Cm * dynP * refArea * refLength;
383  double momZ = store.forces.getCroll() * dynP * refArea * refLength;
384 
385  // Compute acceleration in rocket coordinates
386  store.angularAcceleration = new Coordinate(momX / store.massData.getLongitudinalInertia(),
387  momY / store.massData.getLongitudinalInertia(),
388  momZ / store.massData.getRotationalInertia());
389 
390  store.rollAcceleration = store.angularAcceleration.z;
391  // TODO: LOW: This should be hypot, but does it matter?
392  store.lateralPitchAcceleration = MathUtil.max(Math.abs(store.angularAcceleration.x),
393  Math.abs(store.angularAcceleration.y));
394 
395  store.angularAcceleration = store.thetaRotation.rotateZ(store.angularAcceleration);
396 
397  // Convert to world coordinates
398  store.angularAcceleration = status.getRocketOrientationQuaternion().rotate(store.angularAcceleration);
399 
400  }
401 
402  // Call post-listeners
403  store.accelerationData = SimulationListenerHelper.firePostAccelerationCalculation(status, store.accelerationData);
404  }

411  private void calculateForces(RK4SimulationStatus status, DataStore store) throws SimulationException {

415  if (store.forces != null) {

418

466  private void calculateFlightConditions(RK4SimulationStatus status, DataStore store)
467  throws SimulationException {

470  store.flightConditions = SimulationListenerHelper.firePreFlightConditions(
471  status);
472  if (store.flightConditions != null) {
473  // Compute the store values
474  store.thetaRotation = new Rotation2D(store.flightConditions.getTheta());
475  store.lateralPitchRate = Math.hypot(store.flightConditions.getPitchRate(), store.flightConditions.getYawRate());
476  return;

483  store.flightConditions = new FlightConditions(status.getConfiguration());
484  store.flightConditions.setAtmosphericConditions(atmosphere);
485

488  Coordinate windSpeed = modelWindVelocity(status);
489  Coordinate airSpeed = status.getRocketVelocity().add(windSpeed);
490  airSpeed = status.getRocketOrientationQuaternion().invRotate(airSpeed);
491 
492 
493  // Lateral direction:
494  double len = MathUtil.hypot(airSpeed.x, airSpeed.y);
495  if (len > 0.0001) {
496  store.thetaRotation = new Rotation2D(airSpeed.y / len, airSpeed.x / len);
497  store.flightConditions.setTheta(Math.atan2(airSpeed.y, airSpeed.x));
498  } else {
499  store.thetaRotation = Rotation2D.ID;
500  store.flightConditions.setTheta(0);
501  }
502

504  store.flightConditions.setVelocity(velocity);
505  if (velocity > 0.01) {
506  // aoa must be calculated from the monotonous cosine
507  // sine can be calculated by a simple division
508  store.flightConditions.setAOA(Math.acos(airSpeed.z / velocity), len / velocity);
509  } else {
510  store.flightConditions.setAOA(0);
511  }
512 
513 
514  // Roll, pitch and yaw rate
515  Coordinate rot = status.getRocketOrientationQuaternion().invRotate(status.getRocketRotationVelocity());
516  rot = store.thetaRotation.invRotateZ(rot);
517 
518  store.flightConditions.setRollRate(rot.z);
519  if (len < 0.001) {
520  store.flightConditions.setPitchRate(0);
521  store.flightConditions.setYawRate(0);
522  store.lateralPitchRate = 0;
523  } else {
524  store.flightConditions.setPitchRate(rot.y);
525  store.flightConditions.setYawRate(rot.x);
526  // TODO: LOW: set this as power of two?
527  store.lateralPitchRate = MathUtil.hypot(rot.x, rot.y);
528  }
529 
530 
531  // Call post listeners
532  FlightConditions c = SimulationListenerHelper.firePostFlightConditions(
533  status, store.flightConditions);
534  if (c != store.flightConditions) {
535  // Listeners changed the values, recalculate data store
536  store.flightConditions = c;
537  store.thetaRotation = new Rotation2D(store.flightConditions.getTheta());
538  store.lateralPitchRate = Math.hypot(store.flightConditions.getPitchRate(), store.flightConditions.getYawRate());
539  }
