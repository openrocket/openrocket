113  private String subTimeRanges(String str) {
114 
115  Pattern p = Pattern.compile(variableRegex() + "\\[[^\\]]*:.*?\\]");
116  Matcher m = p.matcher(str);
117 
118  // for each match, make a new custom expression (in subExpressions) with a hashed name
119  // and replace the expression and variable in the original expression string with [hash].
120  while (m.find()) {
121  String match = m.group();
122 
123  int start = match.indexOf("[");
124  int end = match.indexOf("]");
125  int colon = match.indexOf(":");
126 
127  String startTime = match.substring(start + 1, colon);
128  String endTime = match.substring(colon + 1, end);
129  String variableType = match.substring(0, start);
130 
131  RangeExpression exp = new RangeExpression(doc, startTime, endTime, variableType);
132  subExpressions.add(exp);
133  str = str.replace(match, exp.hash());
134  }
