241  private void expandAll(RocketPrintTree theTree, TreePath parent, boolean expand) {
242  theTree.addSelectionPath(parent);
243  // Traverse children
244  TreeNode node = (TreeNode) parent.getLastPathComponent();
245  if (node.getChildCount() >= 0) {
246  for (Enumeration<?> e = node.children(); e.hasMoreElements();) {
247  TreeNode n = (TreeNode) e.nextElement();
248  TreePath path = parent.pathByAddingChild(n);
249  expandAll(theTree, path, expand);
250  }
251  }
252  // Expansion or collapse must be done bottom-up
253  if (expand) {
254  theTree.expandPath(parent);
255  } else {
256  theTree.collapsePath(parent);
257  }
258  }
