80  public boolean handleTouch(MotionEvent event) {
81  int action = event.getAction();
82  if (mRenderer != null && action == MotionEvent.ACTION_MOVE) {
83  if (oldX >= 0 || oldY >= 0) {
84  float newX = event.getX(0);
85  float newY = event.getY(0);
86  if (event.getPointerCount() > 1 && (oldX2 >= 0 || oldY2 >= 0) && mRenderer.isZoomEnabled()) {
87  float newX2 = event.getX(1);
88  float newY2 = event.getY(1);
89  float newDeltaX = Math.abs(newX - newX2);
90  float newDeltaY = Math.abs(newY - newY2);
91  float oldDeltaX = Math.abs(oldX - oldX2);
92  float oldDeltaY = Math.abs(oldY - oldY2);
93  float zoomRate = 1;
94 
95  float tan1 = Math.abs(newY - oldY) / Math.abs(newX - oldX);
96  float tan2 = Math.abs(newY2 - oldY2) / Math.abs(newX2 - oldX2);
97  if ( tan1 <= 0.577 && tan2 <= 0.577) {
98  // horizontal pinch zoom, |deltaY| / |deltaX| is [0 ~ 0.577], 0.577 is the approximate value of tan(Pi/6)
99  zoomRate = newDeltaX / oldDeltaX;
100  if (zoomRate > 0.909 && zoomRate < 1.1) {
101  mPinchZoom.setZoomRate(zoomRate);
102  mPinchZoom.apply(Zoom.ZOOM_AXIS_X);
103  }
104  } else if ( tan1 >= 1.732 && tan2 >= 1.732 ) {
105  // pinch zoom vertically, |deltaY| / |deltaX| is [1.732 ~ infinity], 1.732 is the approximate value of tan(Pi/3)
106  zoomRate = newDeltaY / oldDeltaY;
107  if (zoomRate > 0.909 && zoomRate < 1.1) {
108  mPinchZoom.setZoomRate(zoomRate);
109  mPinchZoom.apply(Zoom.ZOOM_AXIS_Y);
110  }
111  } else if ( (tan1 > 0.577 && tan1 < 1.732) && (tan2 > 0.577 && tan2 < 1.732) ){
112  // pinch zoom diagonally
113  if (Math.abs(newX - oldX) >= Math.abs(newY - oldY)) {
114  zoomRate = newDeltaX / oldDeltaX;
115  } else {
116  zoomRate = newDeltaY / oldDeltaY;
117  }
118  if (zoomRate > 0.909 && zoomRate < 1.1) {
