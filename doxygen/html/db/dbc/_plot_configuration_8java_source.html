438  protected void fitAxes(FlightDataBranch data) {
439 
440  // Reset axes
441  for (Axis a : allAxes) {
442  a.reset();
443  }
444 
445  // Add full range to the axes
446  int length = plotDataTypes.size();
447  for (int i = 0; i < length; i++) {
448  FlightDataType type = plotDataTypes.get(i);
449  Unit unit = plotDataUnits.get(i);
450  int index = plotDataAxes.get(i);
451  if (index < 0) {
452  throw new IllegalStateException("fitAxes called with auto-selected axis");
453  }
454  Axis axis = allAxes.get(index);
455 
456  double min = unit.toUnit(data.getMinimum(type));
457  double max = unit.toUnit(data.getMaximum(type));
458 
459  axis.addBound(min);
460  axis.addBound(max);
461  }
462 
463  // Ensure non-zero (or NaN) range, add a few percent range, include zero if it is close
464  for (Axis a : allAxes) {
465  if (MathUtil.equals(a.getMinValue(), a.getMaxValue())) {
466  a.addBound(a.getMinValue() - 1);
467  a.addBound(a.getMaxValue() + 1);
468  }
469 
470  double addition = a.getRangeLength() * 0.03;
471  a.addBound(a.getMinValue() - addition);
472  a.addBound(a.getMaxValue() + addition);
473 
474  double dist;
475  dist = Math.min(Math.abs(a.getMinValue()), Math.abs(a.getMaxValue()));
476  if (dist <= a.getRangeLength() * INCLUDE_ZERO_DISTANCE) {
477  a.addBound(0);
478  }
479  }
480 
481 
482  // Check whether to use a common zero
483  Axis left = allAxes.get(0);
484  Axis right = allAxes.get(1);
485 
486  if (left.getMinValue() > 0 || left.getMaxValue() < 0 ||
487  right.getMinValue() > 0 || right.getMaxValue() < 0 ||
488  Double.isNaN(left.getMinValue()) || Double.isNaN(right.getMinValue()))
489  return;
490

492 
494  // TODO: MEDIUM: This algorithm may require tweaking
495 
496  double min1 = left.getMinValue();
497  double max1 = left.getMaxValue();
498  double min2 = right.getMinValue();
499  double max2 = right.getMaxValue();

517  min2 = min1;
518  max1 = Math.max(max1, max2);
519  max2 = max1;
520  min2 /= scale;
521  max2 /= scale;
522 
523  // Apply scale

525  left.addBound(max1);
526  right.addBound(min2);
527  right.addBound(max2);

624  protected double getGoodnessValue(FlightDataBranch data) {
625  double goodness = 0;
626  int length = plotDataTypes.size();
627 
628  // Fit the axes ranges to the data
629  fitAxes(data);
630 
631  /*
632  * Calculate goodness of ranges. 100 points is given if the values fill the
633  * entire range, 0 if they fill none of it.
634  */
635  for (int i = 0; i < length; i++) {
636  FlightDataType type = plotDataTypes.get(i);
637  Unit unit = plotDataUnits.get(i);
638  int index = plotDataAxes.get(i);
639  if (index < 0) {
640  throw new IllegalStateException("getGoodnessValue called with auto-selected axis");
641  }
642  Axis axis = allAxes.get(index);
643 
644  double min = unit.toUnit(data.getMinimum(type));
645  double max = unit.toUnit(data.getMaximum(type));
646  if (Double.isNaN(min) || Double.isNaN(max))
647  continue;
648  if (MathUtil.equals(min, max))
649  continue;
650 
651  double d = (max - min) / axis.getRangeLength();
652  d = MathUtil.safeSqrt(d); // Prioritize small ranges
653  goodness += d * 100.0;
654  }
655 
656 
657  /*
658  * Add extra points for specific things.
659  */
660 
661  // A little for the first type being on the first axis
662  if (plotDataAxes.get(0) == 0)
663  goodness += BONUS_FIRST_TYPE_ON_FIRST_AXIS;
664 
665  // A little bonus if the first axis contains zero
666  Axis left = allAxes.get(0);
667  if (left.getMinValue() <= 0 && left.getMaxValue() >= 0)
668  goodness += BONUS_FIRST_AXIS_HAS_ZERO;
669 
670  // A boost if a common zero was used in the ranging
671  Axis right = allAxes.get(1);
672  if (left.getMinValue() <= 0 && left.getMaxValue() >= 0 &&
673  right.getMinValue() <= 0 && right.getMaxValue() >= 0)
674  goodness += BONUS_COMMON_ZERO;
675 
676  // A boost if only one axis is used
677  if (Double.isNaN(left.getMinValue()) || Double.isNaN(right.getMinValue()))
678  goodness += BONUS_ONLY_ONE_AXIS;
679 
680  return goodness;
681  }
