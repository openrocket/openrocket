266  private double paintRocketDiagram(final int thePageImageableWidth, final int thePageImageableHeight,
267  final PdfContentByte theCanvas, final PrintFigure theFigure,
268  final FigureElement theCp, final FigureElement theCg) {
269  theFigure.clearAbsoluteExtra();
270  theFigure.clearRelativeExtra();
271  theFigure.addRelativeExtra(theCp);
272  theFigure.addRelativeExtra(theCg);
273  theFigure.updateFigure();
274 
275  double scale =
276  (thePageImageableWidth * 2.2) / theFigure.getFigureWidth();
277  theFigure.setScale(scale);
278  /*
279  * page dimensions are in points-per-inch, which, in Java2D, are the same as pixels-per-inch; thus we don't need any conversion
280  */
281  theFigure.setSize(thePageImageableWidth, thePageImageableHeight);
282  theFigure.updateFigure();
283 
284  final DefaultFontMapper mapper = new DefaultFontMapper();
285  Graphics2D g2d = theCanvas.createGraphics(thePageImageableWidth, thePageImageableHeight * 2, mapper);
286  final double halfFigureHeight = SCALE_FUDGE_FACTOR * theFigure.getFigureHeightPx() / 2;
287  int y = PrintUnit.POINTS_PER_INCH;
288  //If the y dimension is negative, then it will potentially be drawn off the top of the page. Move the origin
289  //to allow for this.
290  if (theFigure.getDimensions().getY() < 0.0d) {
291  y += (int) halfFigureHeight;
292  }
293  g2d.translate(20, y);
294 
295  g2d.scale(SCALE_FUDGE_FACTOR, SCALE_FUDGE_FACTOR);
296  theFigure.paint(g2d);
297  g2d.dispose();
298  return scale;

399  totalMotorCount += motorCount;
400  totalPropMass += propMass * motorCount;

408  int border = Rectangle.TOP;
409  final PdfPCell motorVCell = ITextHelper.createCell("Total:", border);

416  UnitGroup.UNITS_IMPULSE.getDefaultUnit().toStringUnit(totalImpulse), border));
417  motorTable.addCell(ITextHelper.createCell(
418  ttwFormat.format(totalTTW) + ":1", border));
419  motorTable.addCell(ITextHelper.createCell(
420  UnitGroup.UNITS_MASS.getDefaultUnit().toStringUnit(totalPropMass), border));
421  motorTable.addCell(ITextHelper.createCell("", border));
422 
423  }

508  private FlightData findSimulation(final String motorId, List<Simulation> simulations) {
509  // Perform flight simulation
510  FlightData flight = null;
511  try {

513  Simulation simulation = simulations.get(i);

515  simulation = simulation.copy();
516  simulation.simulate();
517  flight = simulation.getSimulatedData();
518  break;
519  }
520  }
521  } catch (SimulationException e1) {
522  // Ignore
523  }
