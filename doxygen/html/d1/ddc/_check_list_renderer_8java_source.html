72  public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected,
73  boolean cellHasFocus) {
74 
75  setComponentOrientation(list.getComponentOrientation());
76 
77  Color bg = null;
78  Color fg = null;
79 
80  JList.DropLocation dropLocation = list.getDropLocation();
81  if (dropLocation != null && !dropLocation.isInsert() && dropLocation.getIndex() == index) {
82 
83  bg = UIManager.getColor("List.dropCellBackground");
84  fg = UIManager.getColor("List.dropCellForeground");
85 
86  isSelected = true;
87  }
88 
89  if (isSelected) {
90  setBackground(bg == null ? list.getSelectionBackground() : bg);
91  setForeground(fg == null ? list.getSelectionForeground() : fg);
92  } else {
93  setBackground(list.getBackground());
94  setForeground(list.getForeground());
95  }
96 
97  if (value instanceof Icon) {
98  setIcon((Icon) value);
99  setText("");
100  } else {
101  setIcon(null);
102  setText(getObjectAsText(value));
103  }
104 
105  setSelected(isChecked(list, index));
106 
107  setEnabled(list.isEnabled());
108  setFont(list.getFont());
109 
110  Border border = null;
111  if (cellHasFocus) {
112  if (isSelected) {
113  border = UIManager.getBorder("List.focusSelectedCellHighlightBorder");
114  }
115  if (border == null) {
116  border = UIManager.getBorder("List.focusCellHighlightBorder");
117  }
118  } else {
119  border = getNoFocusBorder();
120  }
121  setBorder(border);
122 
123  return this;
124  }
