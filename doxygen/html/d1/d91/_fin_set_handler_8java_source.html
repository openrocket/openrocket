252  @Override
253  public void endHandler(String element, HashMap<String, String> attributes,
254  String content, WarningSet warnings) throws SAXException {
255  //Create the fin set and correct for overrides and actual material densities
256  final FinSet finSet = asOpenRocket(warnings);
257 
258  finSet.setAppearance(appearanceBuilder.getAppearance());
259 
260  if (component.isCompatible(finSet)) {
261  BaseHandler.setOverride(finSet, override, mass, cg);
262  if (!override && finSet.getCrossSection().equals(FinSet.CrossSection.AIRFOIL)) {
263  //Override mass anyway. This is done only for AIRFOIL because Rocksim does not compute different
264  //mass/cg for different cross sections, but OpenRocket does. This can lead to drastic differences
265  //in mass. To counteract that, the cross section value is retained but the mass/cg is overridden
266  //with the calculated values from Rocksim. This will best approximate the Rocksim design in OpenRocket.
267  BaseHandler.setOverride(finSet, true, calcMass, calcCg);
268  }
269  BaseHandler.updateComponentMaterial(finSet, materialName, Material.Type.BULK, density);
270  component.addChild(finSet);
271  }
272  else {
273  warnings.add(finSet.getComponentName() + " can not be attached to "
274  + component.getComponentName() + ", ignoring component.");
275  }
276  }

286  public FinSet asOpenRocket(WarningSet warnings) {
287  FinSet result;
288 
289  if (shapeCode == 0) {
290  //Trapezoidal
291  result = new TrapezoidFinSet();
292  ((TrapezoidFinSet) result).setFinShape(rootChord, tipChord, sweepDistance, semiSpan, thickness);
293  }
294  else if (shapeCode == 1) {
295  //Elliptical
296  result = new EllipticalFinSet();
297  ((EllipticalFinSet) result).setHeight(semiSpan);
298  ((EllipticalFinSet) result).setLength(rootChord);
299  }
300  else if (shapeCode == 2) {
301 
302  result = new FreeformFinSet();
303  try {
304  ((FreeformFinSet) result).setPoints(toCoordinates(pointList, warnings));
305  } catch (IllegalFinPointException e) {
306  warnings.add("Illegal fin point set. " + e.getMessage() + " Ignoring.");
307  }
308  }
309  else {
310  return null;
311  }
312  result.setThickness(thickness);
313  result.setName(name);
314  result.setFinCount(finCount);
315  result.setFinish(finish);
316  //All TTW tabs in Rocksim are relative to the front of the fin.
317  result.setTabRelativePosition(FinSet.TabRelativePosition.FRONT);
318  result.setTabHeight(tabDepth);
319  result.setTabLength(tabLength);
320  result.setTabShift(taboffset);
321  result.setBaseRotation(radialAngle);
322  result.setCrossSection(convertTipShapeCode(tipShapeCode));
323  result.setRelativePosition(position);
324  PositionDependentHandler.setLocation(result, position, location);
325  return result;

338  private Coordinate[] toCoordinates(String pointList, WarningSet warnings) {
339  List<Coordinate> result = new ArrayList<Coordinate>();
340  if (pointList != null && pointList.length() > 0) {
341  String[] points = pointList.split("\\Q|\\E");
342  for (String point : points) {
343  String[] aPoint = point.split(",");
344  try {
345  if (aPoint.length > 1) {
346  Coordinate c = new Coordinate(
347  Double.parseDouble(aPoint[0]) / RocksimCommonConstants.ROCKSIM_TO_OPENROCKET_LENGTH,
348  Double.parseDouble(aPoint[1]) / RocksimCommonConstants.ROCKSIM_TO_OPENROCKET_LENGTH);
349  result.add(c);
350  }
351  else {
352  warnings.add("Invalid fin point pair.");
353  }
354  } catch (NumberFormatException nfe) {
355  warnings.add("Fin point not in numeric format.");
356  }
357  }
358  if (!result.isEmpty()) {
359  //OpenRocket requires fin plan points be ordered from leading root chord to trailing root chord in the
360  //Coordinate array.
361  Coordinate last = result.get(result.size() - 1);
362  if (last.x == 0 && last.y == 0) {
363  Collections.reverse(result);
364  }
365  }
366  }
367  final Coordinate[] coords = new Coordinate[result.size()];
368  return result.toArray(coords);
369  }
