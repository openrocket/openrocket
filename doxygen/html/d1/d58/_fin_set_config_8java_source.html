295  private static double computeFinTabLength(List<CenteringRing> rings, Double finPositionFromTop, Double finLength, DoubleModel mts,
296  final RocketComponent relativeTo) {
297  List<SortableRing> positionsFromTop = new ArrayList<SortableRing>();
298 
299  //Fin tabs will be computed between the last two rings that meet the criteria, represented by top and bottom here.
300  SortableRing top = null;
301  SortableRing bottom = null;
302 
303  if (rings != null) {
304  //Sort rings from top of parent to bottom
305  Collections.sort(rings, new Comparator<CenteringRing>() {
306  @Override
307  public int compare(CenteringRing centeringRing, CenteringRing centeringRing1) {
308  return (int) (1000d * (centeringRing.asPositionValue(RocketComponent.Position.TOP, relativeTo) -
309  centeringRing1.asPositionValue(RocketComponent.Position.TOP, relativeTo)));
310  }
311  });
312 
313  for (int i = 0; i < rings.size(); i++) {
314  CenteringRing centeringRing = rings.get(i);
315  //Handle centering rings that overlap or are adjacent by synthetically merging them into one virtual ring.
316  if (!positionsFromTop.isEmpty() &&
317  positionsFromTop.get(positionsFromTop.size() - 1).bottomSidePositionFromTop() >=
318  centeringRing.asPositionValue(RocketComponent.Position.TOP, relativeTo)) {
319  SortableRing adjacent = positionsFromTop.get(positionsFromTop.size() - 1);
320  adjacent.merge(centeringRing, relativeTo);
321  } else {
322  positionsFromTop.add(new SortableRing(centeringRing, relativeTo));
323  }
324  }
325 
326  for (int i = 0; i < positionsFromTop.size(); i++) {
327  SortableRing sortableRing = positionsFromTop.get(i);
328  if (top == null) {
329  top = sortableRing;
330  } else if (sortableRing.bottomSidePositionFromTop() <= finPositionFromTop) {
331  top = sortableRing;
332  bottom = null;
333  } else if (top.bottomSidePositionFromTop() <= finPositionFromTop) {
334  if (bottom == null) {
335  //If the current ring is in the upper half of the root chord, make it the top ring
336  if (sortableRing.bottomSidePositionFromTop() < finPositionFromTop + finLength / 2d) {
337  top = sortableRing;
338  } else {
339  bottom = sortableRing;
340  }
341  }
342  //Is the ring even with or above the end of the root chord? If so, make the existing bottom the top ring,
343  //and the current ring the bottom
344  else if (sortableRing.positionFromTop() <= finPositionFromTop + finLength) {
345  top = bottom;
346  bottom = sortableRing;
347  }
348  } else {
349  if (bottom == null) {
350  bottom = sortableRing;
351  }
352  }
353  }
354  }
355 
356  double resultFinTabLength = 0d;
357 
358  // Edge case where there are no centering rings or for some odd reason top and bottom are identical.
359  if (top == null || top == bottom) {
360  mts.setValue(0);
361  resultFinTabLength = finLength;
362  } else if (bottom == null) {
363  // If there is no bottom ring and the top ring's bottom edge is within the span of the root chord, then
364  // set the position of the fin tab starting at the bottom side of the top ring.
365  if (top.bottomSidePositionFromTop() >= finPositionFromTop) {
366  mts.setValue(top.bottomSidePositionFromTop() - finPositionFromTop);
367  resultFinTabLength = (finPositionFromTop + finLength - top.bottomSidePositionFromTop());
368  } else {
369  mts.setValue(0);
370  double diffLen = top.positionFromTop() - finPositionFromTop;
371  if (diffLen < 0) {
372  // Otherwise the top ring is outside the span of the root chord so set the tab length to be the entire
373  // root chord.
374  resultFinTabLength = finLength;
375  }
376  else {
377  // Otherwise there is one ring within the span. Return the length from the start of the fin to the top
378  // side of the ring.
379  resultFinTabLength = diffLen;
380  }
381  }
382  }
383  // If the bottom edge of the top centering ring is above the start of the fin's root chord, then make the
384  // fin tab align with the start of the root chord.
385  else if (top.bottomSidePositionFromTop() < finPositionFromTop) {
386  mts.setValue(0);
387 
388  double lenToBottomRing = bottom.positionFromTop - finPositionFromTop;
389  // If the bottom ring lies farther back (down) than the trailing edge of the fin, then the tab should
390  // only be as long as the fin.
391  if (lenToBottomRing > finLength) {
392  resultFinTabLength = finLength;
393  }
394  else {
395  resultFinTabLength = lenToBottomRing;
396  }
397  } else {
398  mts.setValue(top.bottomSidePositionFromTop() - finPositionFromTop);
399  // The bottom ring is beyond the trailing edge of the fin.
400  if (bottom.positionFromTop() > finLength + finPositionFromTop) {
401  resultFinTabLength = (finLength + finPositionFromTop - top.bottomSidePositionFromTop());
402  }
403  // The rings are within the span of the root chord. Place the tab between them.
404  else {
405  resultFinTabLength = (bottom.positionFromTop() - top.bottomSidePositionFromTop());
406  }
407  }
408  if (resultFinTabLength < 0) {
409  resultFinTabLength = 0d;
410  }
411  return resultFinTabLength;
412  }
