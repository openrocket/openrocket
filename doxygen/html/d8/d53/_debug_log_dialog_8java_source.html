502  private class LocationComparator implements Comparator<Object> {
503  private final Pattern splitPattern = Pattern.compile("^\\(([^:]*+):([0-9]++).*\\)$");
504 
505  @Override
506  public int compare(Object o1, Object o2) {
507  String s1 = o1.toString();
508  String s2 = o2.toString();
509 
510  Matcher m1 = splitPattern.matcher(s1);
511  Matcher m2 = splitPattern.matcher(s2);
512 
513  if (m1.matches() && m2.matches()) {
514  String class1 = m1.group(1);
515  String pos1 = m1.group(2);
516  String class2 = m2.group(1);
517  String pos2 = m2.group(2);
518 
519  if (class1.equals(class2)) {
520  return NumericComparator.INSTANCE.compare(pos1, pos2);
521  } else {
522  return class1.compareTo(class2);
523  }
524  }
525 
526  return s1.compareTo(s2);
527  }
