197  // The <Ring> XML element in Rocksim design file is used for many types of components, unfortunately.
198  // Additional subelements are used to indicate the type of the rocket component. When parsing using SAX
199  // this poses a problem because we can't "look ahead" to see what type is being represented at the start
200  // of parsing - something that would be nice to do so that we can instantiate the correct OR component
201  // at the start, then just call setters for the appropriate data.
202 
203  // To overcome that, a CenteringRing is instantiated at the start of parsing, it's mutators are called,
204  // and then at the end (this method) converts the CenteringRing to a more appropriate type. CenteringRing
205  // is generic enough to support the representation of all similar types without loss of data.
206 
207  //UsageCode
208  // 0 == Centering Ring
209  // 1 == Bulkhead
210  // 2 == Engine Block
211  // 3 == Sleeve
212  // 4 == Tube Coupler
213 
214  if (usageCode == 1) {
215  //Bulkhead
216  asBulkhead(warnings);
217  } else if (usageCode == 2) {
218  asEngineBlock(warnings);
219  } else if (usageCode == 4) {
220  //TubeCoupler
221  asTubeCoupler(warnings);
222  } else {
223  //Default
224  asCenteringRing(warnings);
225  }
