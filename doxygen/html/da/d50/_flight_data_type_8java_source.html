262  public static synchronized FlightDataType getType(String s, String symbol, UnitGroup u) {
263 
264  // if symbol is null : try finding by name
265  // if unit is null : don't do anything to the unit if found, just return datatype if found and generate an error and an empty unit otherwise
266  int oldPriority = DEFAULT_PRIORITY;
267 
268  //FlightDataType type = findFromSymbol(symbol);
269  FlightDataType type = EXISTING_TYPES.get(symbol);
270 
271  if (type != null) {
272  // found it from symbol
273 
274  // if name was not given (empty string), can use the one we found
275  if ( s == null || StringUtil.isEmpty(s)){
276  s = type.getName();
277  }
278  if ( u == null ){
279  u = type.getUnitGroup();
280  }
281 
282  // if something has changed, then we need to remove the old one
283  // otherwise, just return what we found
284  if ( !u.equals(type.getUnitGroup()) )
285  {
286  oldPriority = type.priority;
287  EXISTING_TYPES.remove(type);
288  log.info("Unitgroup of type "+type.getName() +
289  ", has changed from "+type.getUnitGroup().toString() +
290  " to "+u.toString() +
291  ". Removing old version.");
292  }
293  else if (!s.equals(type.getName())) {
294  oldPriority = type.priority;
295  EXISTING_TYPES.remove(type);
296  log.info("Name of type "+type.getName()+", has changed to "+s+". Removing old version.");
297  }
298  else{
299  return type;
300  }
301  }
302 
303  if (u == null){
304  u = UnitGroup.UNITS_NONE;
305  log.error("Made a new flightdatatype, but did not know what units to use.");
306  }
307 
308  // make a new one
309  type = newType(s, symbol, u, oldPriority);
310  return type;
311  }
