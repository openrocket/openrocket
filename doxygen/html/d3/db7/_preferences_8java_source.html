702  // Search preferences
703  Class<?> c = componentClass;
704  while (c != null && RocketComponent.class.isAssignableFrom(c)) {
705  String value = this.getString(directory, c.getSimpleName(), null);
706  if (value != null)
707  return value;

710

712  return null;

714  // Search defaults

718  if (value != null)
719  return value;
720  c = c.getSuperclass();
721  }

723  return null;
