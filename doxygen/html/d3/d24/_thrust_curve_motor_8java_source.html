301 
302  // Maximum thrust
303  maxThrust = 0;
304  for (double t : thrust) {
305  if (t > maxThrust)
306  maxThrust = t;
307  }
308 
309 
310  // Burn start time
311  double thrustLimit = maxThrust * MARGINAL_THRUST;
312  double burnStart, burnEnd;
313 
314  int pos;
315  for (pos = 1; pos < thrust.length; pos++) {
316  if (thrust[pos] >= thrustLimit)
317  break;
318  }
319  if (pos >= thrust.length) {
320  throw new BugException("Could not compute burn start time, maxThrust=" + maxThrust +
321  " limit=" + thrustLimit + " thrust=" + Arrays.toString(thrust));
322  }
323  if (MathUtil.equals(thrust[pos - 1], thrust[pos])) {
324  // For safety
325  burnStart = (time[pos - 1] + time[pos]) / 2;
326  } else {
327  burnStart = MathUtil.map(thrustLimit, thrust[pos - 1], thrust[pos], time[pos - 1], time[pos]);
328  }
329 
330 
331  // Burn end time
332  for (pos = thrust.length - 2; pos >= 0; pos--) {
333  if (thrust[pos] >= thrustLimit)
334  break;
335  }
336  if (pos < 0) {
337  throw new BugException("Could not compute burn end time, maxThrust=" + maxThrust +
338  " limit=" + thrustLimit + " thrust=" + Arrays.toString(thrust));
339  }
340  if (MathUtil.equals(thrust[pos], thrust[pos + 1])) {
341  // For safety
342  burnEnd = (time[pos] + time[pos + 1]) / 2;
343  } else {
344  burnEnd = MathUtil.map(thrustLimit, thrust[pos], thrust[pos + 1],
345  time[pos], time[pos + 1]);
346  }
347 
348 
349  // Burn time
350  burnTime = Math.max(burnEnd - burnStart, 0);
351 
352 
353  // Total impulse and average thrust
354  totalImpulse = 0;
355  averageThrust = 0;
356 
357  for (pos = 0; pos < time.length - 1; pos++) {
358  double t0 = time[pos];
359  double t1 = time[pos + 1];
360  double f0 = thrust[pos];
361  double f1 = thrust[pos + 1];
362 
363  totalImpulse += (t1 - t0) * (f0 + f1) / 2;
364 
365  if (t0 < burnStart && t1 > burnStart) {
366  double fStart = MathUtil.map(burnStart, t0, t1, f0, f1);
367  averageThrust += (fStart + f1) / 2 * (t1 - burnStart);
368  } else if (t0 >= burnStart && t1 <= burnEnd) {
369  averageThrust += (f0 + f1) / 2 * (t1 - t0);
370  } else if (t0 < burnEnd && t1 > burnEnd) {
371  double fEnd = MathUtil.map(burnEnd, t0, t1, f0, f1);
372  averageThrust += (f0 + fEnd) / 2 * (burnEnd - t0);
373  }
374  }
375 
376  if (burnTime > 0) {
377  averageThrust /= burnTime;
378  } else {
379  averageThrust = 0;
380  }

512

514  if (nextTime < time[position + 1]) {
515

518  thrust[position], thrust[position + 1]);

528  position++;

530  stepThrust += (thrust[position] + thrust[position + 1]) / 2 *
531  (time[position + 1] - time[position]);
532  position++;
533  }
534 
535  // End step

538  thrust[position], thrust[position + 1]);
539  stepThrust += (thrust[position] + instThrust) / 2 *
540  (nextTime - time[position]);
541  } else {
542  // Thrust ended during this step
543  instThrust = 0;
544  }

548  }
