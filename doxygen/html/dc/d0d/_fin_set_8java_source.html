460  public double getFilletVolume() {
461  /*
462  * Here is how the volume of the fillet is found. It assumes a circular concave
463  * fillet tangent to the fin and the body tube.
464  *
465  * 1. Form a triangle with vertices at the BT center, the tangent point between
466  * the fillet and the fin, and the center of the fillet radius.
467  * 2. The line between the center of the BT and the center of the fillet radius
468  * will pass through the tangent point between the fillet and the BT.
469  * 3. Find the area of the triangle, then subtract the portion of the BT and
470  * fillet that is in that triangle. (angle/2PI * pi*r^2= angle/2 * r^2)
471  * 4. Multiply the remaining area by the length.
472  * 5. Return twice that since there is a fillet on each side of the fin.
473  *
474  */
475  double btRadius = 1000.0; // assume a really big body tube if we can't get the radius,
476  RocketComponent c = this.getParent();
477  if (BodyTube.class.isInstance(c)) {
478  btRadius = ((BodyTube) c).getOuterRadius();
479  }
480  double totalRad = filletRadius + btRadius;
481  double innerAngle = Math.asin(filletRadius / totalRad);
482  double outerAngle = Math.acos(filletRadius / totalRad);
483 
484  double outerArea = Math.tan(outerAngle) * filletRadius * filletRadius / 2;
485  double filletVolume = length * (outerArea
486  - outerAngle * filletRadius * filletRadius / 2
487  - innerAngle * btRadius * btRadius / 2);
488  return 2 * filletVolume;
489  }

491  private void calculateAreaCG() {
492  Coordinate[] points = this.getFinPoints();
493  finArea = 0;
494  finCGx = 0;
495  finCGy = 0;
496 
497  for (int i = 0; i < points.length - 1; i++) {
498  final double x0 = points[i].x;
499  final double x1 = points[i + 1].x;
500  final double y0 = points[i].y;
501  final double y1 = points[i + 1].y;
502 
503  double da = (y0 + y1) * (x1 - x0) / 2;
504  finArea += da;
505  if (Math.abs(y0 + y1) < 0.00001) {
506  finCGx += (x0 + x1) / 2 * da;
507  finCGy += y0 / 2 * da;
508  } else {
509  finCGx += (x0 * (2 * y0 + y1) + x1 * (y0 + 2 * y1)) / (3 * (y0 + y1)) * da;
510  finCGy += (y1 + y0 * y0 / (y0 + y1)) / 3 * da;
511  }
512  }
513 
514  if (finArea < 0)
515  finArea = 0;
516 
517  // Add effect of fin tabs to CG
518  double tabArea = tabLength * tabHeight;
519  if (!MathUtil.equals(tabArea, 0)) {
520 
521  double x = (getTabFrontEdge() + getTabTrailingEdge()) / 2;
522  double y = -this.tabHeight / 2;
523 
524  finCGx += x * tabArea;
525  finCGy += y * tabArea;
526 
527  }
528 
529  if ((finArea + tabArea) > 0) {
530  finCGx /= (finArea + tabArea);
531  finCGy /= (finArea + tabArea);
532  } else {
533  finCGx = (points[0].x + points[points.length - 1].x) / 2;
534  finCGy = 0;
535  }
536  }

539  /*
540  * Return an approximation of the longitudinal unitary inertia of the fin set.
541  * The process is the following:
542  *
543  * 1. Approximate the fin with a rectangular fin
544  *
545  * 2. The inertia of one fin is taken as the average of the moments of inertia
546  * through its center perpendicular to the plane, and the inertia through
547  * its center parallel to the plane
548  *
549  * 3. If there are multiple fins, the inertia is shifted to the center of the fin
550  * set and multiplied by the number of fins.
551  */
552  @Override
553  public double getLongitudinalUnitInertia() {
554  double area = getFinArea();
555  if (MathUtil.equals(area, 0))
556  return 0;
557 
558  // Approximate fin with a rectangular fin
559  // w2 and h2 are squares of the fin width and height
560  double w = getLength();
561  double h = getSpan();
562  double w2, h2;
563 
564  if (MathUtil.equals(w * h, 0)) {
565  w2 = area;
566  h2 = area;
567  } else {
568  w2 = w * area / h;
569  h2 = h * area / w;
570  }
571 
572  double inertia = (h2 + 2 * w2) / 24;
573 
574  if (fins == 1)
575  return inertia;
576 
577  double radius = getBodyRadius();
578 
579  return fins * (inertia + MathUtil.pow2(MathUtil.safeSqrt(h2) + radius));
580  }

583  /*
584  * Return an approximation of the rotational unitary inertia of the fin set.
585  * The process is the following:
586  *
587  * 1. Approximate the fin with a rectangular fin and calculate the inertia of the
588  * rectangular approximate
589  *
590  * 2. If there are multiple fins, shift the inertia center to the fin set center
591  * and multiply with the number of fins.
592  */
593  @Override
594  public double getRotationalUnitInertia() {
595  double area = getFinArea();
596  if (MathUtil.equals(area, 0))
597  return 0;
598 
599  // Approximate fin with a rectangular fin
600  double w = getLength();
601  double h = getSpan();
602 
603  if (MathUtil.equals(w * h, 0)) {
604  h = MathUtil.safeSqrt(area);
605  } else {
606  h = MathUtil.safeSqrt(h * area / w);
607  }
608 
609  if (fins == 1)
610  return h * h / 12;
611 
612  double radius = getBodyRadius();
613 
614  return fins * (h * h / 12 + MathUtil.pow2(h / 2 + radius));
