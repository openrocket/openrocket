107  public boolean tryShow() {
108  mPopup = new IcsListPopupWindow(mContext, null, R.attr.popupMenuStyle);
109  mPopup.setOnDismissListener(this);

112  mAdapter = new MenuAdapter(mMenu);
113  mPopup.setAdapter(mAdapter);
114  mPopup.setModal(true);

116  View anchor = mAnchorView;
117  if (anchor != null) {
118  final boolean addGlobalListener = mTreeObserver == null;
119  mTreeObserver = anchor.getViewTreeObserver(); // Refresh to latest
120  if (addGlobalListener) mTreeObserver.addOnGlobalLayoutListener(this);
121  ((View_HasStateListenerSupport)anchor).addOnAttachStateChangeListener(this);
122  mPopup.setAnchorView(anchor);

128  mPopup.setInputMethodMode(PopupWindow.INPUT_METHOD_NOT_NEEDED);

169  private int measureContentWidth(ListAdapter adapter) {
170  // Menus don't tend to be long, so this is more sane than it looks.
171  int width = 0;
172  View itemView = null;
173  int itemType = 0;
174  final int widthMeasureSpec =
175  MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED);
176  final int heightMeasureSpec =
177  MeasureSpec.makeMeasureSpec(0, MeasureSpec.UNSPECIFIED);
178  final int count = adapter.getCount();
179  for (int i = 0; i < count; i++) {
180  final int positionType = adapter.getItemViewType(i);
181  if (positionType != itemType) {
182  itemType = positionType;
183  itemView = null;
184  }
185  if (mMeasureParent == null) {
186  mMeasureParent = new FrameLayout(mContext);
187  }
188  itemView = adapter.getView(i, itemView, mMeasureParent);
189  itemView.measure(widthMeasureSpec, heightMeasureSpec);
190  width = Math.max(width, itemView.getMeasuredWidth());
191  }
192  return width;
193  }

243  if (subMenu.hasVisibleItems()) {
244  MenuPopupHelper subPopup = new MenuPopupHelper(mContext, subMenu, mAnchorView, false);
245  subPopup.setCallback(mPresenterCallback);
246 
247  boolean preserveIconSpacing = false;
248  final int count = subMenu.size();
249  for (int i = 0; i < count; i++) {
250  MenuItem childItem = subMenu.getItem(i);
251  if (childItem.isVisible() && childItem.getIcon() != null) {
252  preserveIconSpacing = true;
253  break;
254  }
255  }
256  subPopup.setForceShowIcon(preserveIconSpacing);
257 
258  if (subPopup.tryShow()) {
259  if (mPresenterCallback != null) {
260  mPresenterCallback.onOpenSubMenu(subMenu);
261  }
262  return true;
263  }
264  }
265  return false;
266  }

353  void findExpandedIndex() {
354  final MenuItemImpl expandedItem = mMenu.getExpandedItem();
355  if (expandedItem != null) {
356  final ArrayList<MenuItemImpl> items = mMenu.getNonActionItems();
357  final int count = items.size();
358  for (int i = 0; i < count; i++) {
359  final MenuItemImpl item = items.get(i);
360  if (item == expandedItem) {
361  mExpandedIndex = i;
362  return;
363  }
364  }
365  }
366  mExpandedIndex = -1;
367  }
