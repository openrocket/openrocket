483  private Bounds calculateBounds() {
484  if (cachedBounds != null) {
485  return cachedBounds;
486  } else {
487  final Bounds b = new Bounds();
488  final Collection<Coordinate> bounds = configuration.getBounds();
489  for (Coordinate c : bounds) {
490  b.xMax = Math.max(b.xMax, c.x);
491  b.xMin = Math.min(b.xMin, c.x);
492 
493  b.yMax = Math.max(b.yMax, c.y);
494  b.yMin = Math.min(b.yMin, c.y);
495 
496  b.zMax = Math.max(b.zMax, c.z);
497  b.zMin = Math.min(b.zMin, c.z);
498 
499  double r = MathUtil.hypot(c.y, c.z);
500  b.rMax = Math.max(b.rMax, r);
501  }
502  b.xSize = b.xMax - b.xMin;
503  b.ySize = b.yMax - b.yMin;
504  b.zSize = b.zMax - b.zMin;
505  cachedBounds = b;
506  return b;
507  }

510  private void setupView(final GL2 gl, final GLU glu) {
511  gl.glLoadIdentity();
512 
513  gl.glLightfv(GLLightingFunc.GL_LIGHT1, GLLightingFunc.GL_POSITION,
514  lightPosition, 0);
515 
516  // Get the bounds
517  final Bounds b = calculateBounds();
518 
519  // Calculate the distance needed to fit the bounds in both the X and Y
520  // direction
521  // Add 10% for space around it.
522  final double dX = (b.xSize * 1.2 / 2.0)
523  / Math.tan(Math.toRadians(fovX / 2.0));
524  final double dY = (b.rMax * 2.0 * 1.2 / 2.0)
525  / Math.tan(Math.toRadians(fovY / 2.0));
526 
527  // Move back the greater of the 2 distances
528  glu.gluLookAt(0, 0, Math.max(dX, dY), 0, 0, 0, 0, 1, 0);
529 
530  gl.glRotated(yaw * (180.0 / Math.PI), 0, 1, 0);
531  gl.glRotated(roll * (180.0 / Math.PI), 1, 0, 0);
532 
533  // Center the rocket in the view.
534  gl.glTranslated(-b.xMin - b.xSize / 2.0, 0, 0);
535 
536  //Change to LEFT Handed coordinates
537  gl.glScaled(1, 1, -1);
538  gl.glFrontFace(GL.GL_CW);
539 
540  //Flip textures for LEFT handed coords
541  gl.glMatrixMode(GL.GL_TEXTURE);
542  gl.glLoadIdentity();
543  gl.glScaled(-1, 1, 1);
544  gl.glTranslated(-1, 0, 0);
545  gl.glMatrixMode(GLMatrixFunc.GL_MODELVIEW);
546  }

605  private Coordinate project(final Coordinate c, final GL2 gl, final GLU glu) {
606  final double[] mvmatrix = new double[16];
607  final double[] projmatrix = new double[16];
608  final int[] viewport = new int[4];
609 
610  gl.glGetIntegerv(GL.GL_VIEWPORT, viewport, 0);
611  gl.glGetDoublev(GLMatrixFunc.GL_MODELVIEW_MATRIX, mvmatrix, 0);
612  gl.glGetDoublev(GLMatrixFunc.GL_PROJECTION_MATRIX, projmatrix, 0);
613 
614  final double out[] = new double[4];
615  glu.gluProject(c.x, c.y, c.z, mvmatrix, 0, projmatrix, 0, viewport, 0,
616  out, 0);
617 
618  return new Coordinate(out[0], out[1], out[2]);
