202  @SuppressWarnings("unchecked")
203  private <T> T invoke(Class<T> retType, T def, String method, Object... args) throws SimulationException {
204  try {
205  if (!missing.contains(method)) {
206  Object o = invocable.invokeFunction(method, args);
207  if (o == null) {
208  // Use default/null if function returns nothing
209  return def;
210  } else if (!o.getClass().equals(retType)) {
211  throw new SimulationListenerException("Custom script function " + method + " returned type " +
212  o.getClass().getSimpleName() + ", expected " + retType.getSimpleName());
213  } else {
214  return (T) o;
215  }
216  }
217  } catch (NoSuchMethodException e) {
218  missing.add(method);
219  // fall-through
220  } catch (ScriptException e) {
221  logger.warn("Script exception in " + method + ": " + e, e);
222  throw new SimulationException("Script failed: " + e.getMessage());
223  }
224  return def;
225  }
