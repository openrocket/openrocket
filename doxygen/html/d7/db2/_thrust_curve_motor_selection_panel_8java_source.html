448  List<ThrustCurveMotor> getFilteredCurves() {
449  List<ThrustCurveMotor> motors = selectedMotorSet.getMotors();
450  if (hideSimilarBox.isSelected() && selectedMotor != null) {
451  List<ThrustCurveMotor> filtered = new ArrayList<ThrustCurveMotor>(motors.size());
452  for (int i = 0; i < motors.size(); i++) {
453  ThrustCurveMotor m = motors.get(i);
454  if (m.equals(selectedMotor)) {
455  filtered.add(m);
456  continue;
457  }
458  double similarity = MotorCorrelation.similarity(selectedMotor, m);
459  log.debug("Motor similarity: " + similarity);
460  if (similarity < MOTOR_SIMILARITY_THRESHOLD) {
461  filtered.add(m);
462  }
463  }
464  motors = filtered;
465  }
466 
467  Collections.sort(motors, MOTOR_COMPARATOR);
468 
469  return motors;
