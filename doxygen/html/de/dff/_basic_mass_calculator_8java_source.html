266  private MassData calculateAssemblyMassData(RocketComponent parent) {
267  MassData parentData = new MassData();
268 
269  // Calculate data for this component
270  parentData.cg = parent.getComponentCG();
271  if (parentData.cg.weight < MIN_MASS)
272  parentData.cg = parentData.cg.setWeight(MIN_MASS);
273 
274 
275  // Override only this component's data
276  if (!parent.getOverrideSubcomponents()) {
277  if (parent.isMassOverridden())
278  parentData.cg = parentData.cg.setWeight(MathUtil.max(parent.getOverrideMass(), MIN_MASS));
279  if (parent.isCGOverridden())
280  parentData.cg = parentData.cg.setXYZ(parent.getOverrideCG());
281  }
282 
283  parentData.longitudinalInertia = parent.getLongitudinalUnitInertia() * parentData.cg.weight;
284  parentData.rotationalInetria = parent.getRotationalUnitInertia() * parentData.cg.weight;
285 
286 
287  // Combine data for subcomponents
288  for (RocketComponent sibling : parent.getChildren()) {
289  Coordinate combinedCG;
290  double dx2, dr2;
291 
292  // Compute data of sibling
293  MassData siblingData = calculateAssemblyMassData(sibling);
294  Coordinate[] siblingCGs = sibling.toRelative(siblingData.cg, parent);
295 
296  for (Coordinate siblingCG : siblingCGs) {
297 
298  // Compute CG of this + sibling
299  combinedCG = parentData.cg.average(siblingCG);
300 
301  // Add effect of this CG change to parent inertia
302  dx2 = pow2(parentData.cg.x - combinedCG.x);
303  parentData.longitudinalInertia += parentData.cg.weight * dx2;
304 
305  dr2 = pow2(parentData.cg.y - combinedCG.y) + pow2(parentData.cg.z - combinedCG.z);
306  parentData.rotationalInetria += parentData.cg.weight * dr2;
307 
308 
309  // Add inertia of sibling
310  parentData.longitudinalInertia += siblingData.longitudinalInertia;
311  parentData.rotationalInetria += siblingData.rotationalInetria;
312 
313  // Add effect of sibling CG change
314  dx2 = pow2(siblingData.cg.x - combinedCG.x);
315  parentData.longitudinalInertia += siblingData.cg.weight * dx2;
316 
317  dr2 = pow2(siblingData.cg.y - combinedCG.y) + pow2(siblingData.cg.z - combinedCG.z);
318  parentData.rotationalInetria += siblingData.cg.weight * dr2;
319 
320  // Set combined CG
321  parentData.cg = combinedCG;
322  }
323  }

326  if (parent.getOverrideSubcomponents()) {

329  double newMass = MathUtil.max(parent.getOverrideMass(), MIN_MASS);
330  parentData.longitudinalInertia = parentData.longitudinalInertia * newMass / oldMass;
331  parentData.rotationalInetria = parentData.rotationalInetria * newMass / oldMass;
332  parentData.cg = parentData.cg.setWeight(newMass);
