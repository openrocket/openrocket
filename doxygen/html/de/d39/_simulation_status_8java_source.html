102  // Initialize to roll angle with least stability w.r.t. the wind
103  Quaternion o;
104  FlightConditions cond = new FlightConditions(this.configuration);
105  this.simulationConditions.getAerodynamicCalculator().getWorstCP(this.configuration, cond, null);
106  double angle = -cond.getTheta() - (Math.PI / 2.0 - this.simulationConditions.getLaunchRodDirection());
107  o = Quaternion.rotation(new Coordinate(0, 0, angle));
108 
109  // Launch rod angle and direction
110  o = o.multiplyLeft(Quaternion.rotation(new Coordinate(0, this.simulationConditions.getLaunchRodAngle(), 0)));
111  o = o.multiplyLeft(Quaternion.rotation(new Coordinate(0, 0, Math.PI / 2.0 - this.simulationConditions.getLaunchRodDirection())));
112 
113  this.orientation = o;
114  this.rotationVelocity = Coordinate.NUL;
115 
116  /*
117  * Calculate the effective launch rod length taking into account launch lugs.
118  * If no lugs are found, assume a tower launcher of full length.
119  */
120  double length = this.simulationConditions.getLaunchRodLength();
121  double lugPosition = Double.NaN;
122  for (RocketComponent c : this.configuration) {
123  if (c instanceof LaunchLug) {
124  double pos = c.toAbsolute(new Coordinate(c.getLength()))[0].x;
125  if (Double.isNaN(lugPosition) || pos > lugPosition) {
126  lugPosition = pos;
127  }
128  }
129  }
130  if (!Double.isNaN(lugPosition)) {
131  double maxX = 0;
132  for (Coordinate c : this.configuration.getBounds()) {
133  if (c.x > maxX)
134  maxX = c.x;
135  }
136  if (maxX >= lugPosition) {
137  length = Math.max(0, length - (maxX - lugPosition));
138  }
139  }
140  this.effectiveLaunchRodLength = length;
