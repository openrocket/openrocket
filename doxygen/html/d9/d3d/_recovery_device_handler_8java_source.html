69  protected double computeDensity(RocksimDensityType type, double rawDensity) {
70 
71  double result;
72 
73  if (rawDensity > 0d) {
74  //ROCKSIM_SURFACE is a square area density; compute normally
75  //ROCKSIM_LINE is a single length dimension (kg/m) but Rocksim ignores thickness for this type and treats
76  //it like a SURFACE.
77  if (RocksimDensityType.ROCKSIM_SURFACE.equals(type) || RocksimDensityType.ROCKSIM_LINE.equals(type)) {
78  result = rawDensity / RocksimDensityType.ROCKSIM_SURFACE.asOpenRocket();
79  }
80  //ROCKSIM_BULK is a cubic area density; multiple by thickness to make per square area; the result, when
81  //multiplied by the area will then equal Rocksim's computed mass.
82  else {
83  result = (rawDensity / type.asOpenRocket()) * thickness;
84  }
85  }
86  else {
87  result = calcMass / getComponent().getArea();
88  //A Rocksim bug on streamers/parachutes results in a 0 density at times. When that is detected, try
89  //to compute an approximate density from Rocksim's computed mass.
90  if (RocksimDensityType.ROCKSIM_BULK.equals(type)) {
91  //ROCKSIM_BULK is a cubic area density; multiple by thickness to make per square area
92  result *= thickness;
93  }
94  }
