421  protected double calculateFinCNa1(FlightConditions conditions) {
422  double mach = conditions.getMach();
423  double ref = conditions.getRefArea();
424  double alpha = MathUtil.min(conditions.getAOA(),
425  Math.PI - conditions.getAOA(), STALL_ANGLE);
426 
427  // Subsonic case
428  if (mach <= CNA_SUBSONIC) {
429  return 2 * Math.PI * pow2(span) / (1 + MathUtil.safeSqrt(1 + (1 - pow2(mach)) *
430  pow2(pow2(span) / (finArea * cosGamma)))) / ref;
431  }
432 
433  // Supersonic case
434  if (mach >= CNA_SUPERSONIC) {
435  return finArea * (K1.getValue(mach) + K2.getValue(mach) * alpha +
436  K3.getValue(mach) * pow2(alpha)) / ref;
437  }
438 
439  // Transonic case, interpolate
440  double subV, superV;
441  double subD, superD;
442 
443  double sq = MathUtil.safeSqrt(1 + (1 - pow2(CNA_SUBSONIC)) * pow2(span * span / (finArea * cosGamma)));
444  subV = 2 * Math.PI * pow2(span) / ref / (1 + sq);
445  subD = 2 * mach * Math.PI * pow(span, 6) / (pow2(finArea * cosGamma) * ref *
446  sq * pow2(1 + sq));
447 
448  superV = finArea * (K1.getValue(CNA_SUPERSONIC) + K2.getValue(CNA_SUPERSONIC) * alpha +
449  K3.getValue(CNA_SUPERSONIC) * pow2(alpha)) / ref;
450  superD = -finArea / ref * 2 * CNA_SUPERSONIC / CNA_SUPERSONIC_B;
451 
452  // System.out.println("subV="+subV+" superV="+superV+" subD="+subD+" superD="+superD);
453 
454  return cnaInterpolator.interpolate(mach, subV, superV, subD, superD, 0);
455  }

457  private double calculateDampingMoment(FlightConditions conditions) {
458  double rollRate = conditions.getRollRate();
459 
460  if (Math.abs(rollRate) < 0.1)
461  return 0;
462 
463  double mach = conditions.getMach();
464  double absRate = Math.abs(rollRate);
465 
466  /*
467  * At low speeds and relatively large roll rates (i.e. near apogee) the
468  * fin tips rotate well above stall angle. In this case sum the chords
469  * separately.
470  */
471  if (absRate * (bodyRadius + span) / conditions.getVelocity() > 15 * Math.PI / 180) {
472  double sum = 0;
473  for (int i = 0; i < DIVISIONS; i++) {
474  double dist = bodyRadius + span * i / DIVISIONS;
475  double aoa = Math.min(absRate * dist / conditions.getVelocity(), 15 * Math.PI / 180);
476  sum += chordLength[i] * dist * aoa;
477  }
478  sum = sum * (span / DIVISIONS) * 2 * Math.PI / conditions.getBeta() /
479  (conditions.getRefArea() * conditions.getRefLength());
480 
481  // System.out.println("SPECIAL: " +
482  // (MathUtil.sign(rollRate) *component.getFinCount() * sum));
483  return MathUtil.sign(rollRate) * finCount * sum;
484  }
485 
486  if (mach <= CNA_SUBSONIC) {
487  // System.out.println("BASIC: "+
488  // (component.getFinCount() * 2*Math.PI * rollRate * rollSum /
489  // (conditions.getRefArea() * conditions.getRefLength() *
490  // conditions.getVelocity() * conditions.getBeta())));
491 
492  return finCount * 2 * Math.PI * rollRate * rollSum /
493  (conditions.getRefArea() * conditions.getRefLength() *
494  conditions.getVelocity() * conditions.getBeta());
495  }
496  if (mach >= CNA_SUPERSONIC) {
497 
498  double vel = conditions.getVelocity();
499  double k1 = K1.getValue(mach);
500  double k2 = K2.getValue(mach);
501  double k3 = K3.getValue(mach);
502 
503  double sum = 0;
504 
505  for (int i = 0; i < DIVISIONS; i++) {
506  double y = i * span / (DIVISIONS - 1);
507  double angle = rollRate * (bodyRadius + y) / vel;
508 
509  sum += (k1 * angle + k2 * angle * angle + k3 * angle * angle * angle)
510  * chordLength[i] * (bodyRadius + y);
511  }
512 
513  return finCount * sum * span / (DIVISIONS - 1) /
514  (conditions.getRefArea() * conditions.getRefLength());
515  }
516 
517  // Transonic, do linear interpolation
518 
519  FlightConditions cond = conditions.clone();
520  cond.setMach(CNA_SUBSONIC - 0.01);
521  double subsonic = calculateDampingMoment(cond);
522  cond.setMach(CNA_SUPERSONIC + 0.01);
523  double supersonic = calculateDampingMoment(cond);
524 
525  return subsonic * (CNA_SUPERSONIC - mach) / (CNA_SUPERSONIC - CNA_SUBSONIC) +
526  supersonic * (mach - CNA_SUBSONIC) / (CNA_SUPERSONIC - CNA_SUBSONIC);
527  }

624  public double calculatePressureDragForce(FlightConditions conditions,
625  double stagnationCD, double baseCD, WarningSet warnings) {
626 
627  double mach = conditions.getMach();
628  double drag = 0;
629 
630  // Pressure fore-drag
631  if (crossSection == FinSet.CrossSection.AIRFOIL ||
632  crossSection == FinSet.CrossSection.ROUNDED) {
633 
634  // Round leading edge
635  if (mach < 0.9) {
636  drag = Math.pow(1 - pow2(mach), -0.417) - 1;
637  } else if (mach < 1) {
638  drag = 1 - 1.785 * (mach - 0.9);
639  } else {
640  drag = 1.214 - 0.502 / pow2(mach) + 0.1095 / pow2(pow2(mach));
641  }
642 
643  } else if (crossSection == FinSet.CrossSection.SQUARE) {
644  drag = stagnationCD;
645  } else {
646  throw new UnsupportedOperationException("Unsupported fin profile: " + crossSection);
647  }
648 
649  // Slanted leading edge
650  drag *= pow2(cosGammaLead);
651 
652  // Trailing edge drag
653  if (crossSection == FinSet.CrossSection.SQUARE) {
654  drag += baseCD;
655  } else if (crossSection == FinSet.CrossSection.ROUNDED) {
