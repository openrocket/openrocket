266  private void updateExponentialParameters() {
267  double pos = this.linearPosition;
268  double myMinValue = this.min.getValue();
269  double myMidValue = this.mid.getValue();
270  double myMaxValue = this.max.getValue();
271  /*
272  * quad2..0 are calculated such that
273  * f(pos) = mid - continuity
274  * f(1) = max - end point
275  * f'(pos) = linear1 - continuity of derivative
276  */
277  double delta = (myMidValue - myMinValue) / pos;
278  quad2 = (myMaxValue - myMidValue - delta + delta * pos) / pow2(pos - 1);
279  quad1 = (delta + 2 * (myMidValue - myMaxValue) * pos - delta * pos * pos) / pow2(pos - 1);
280  quad0 = (myMidValue - (2 * myMidValue + delta) * pos + (myMaxValue + delta) * pos * pos) / pow2(pos - 1);
281  }
