300  public InteractiveSimulationWorker(OpenRocketDocument doc, Simulation sim, int index) {
301  super(sim);
302  List<CustomExpression> exprs = doc.getCustomExpressions();
303  exprListener = new CustomExpressionSimulationListener(exprs);
304  this.index = index;
305 
306  // Calculate estimate of motor burn time
307  double launchBurn = 0;
308  double otherBurn = 0;
309  Configuration config = simulation.getConfiguration();
310  String id = simulation.getOptions().getMotorConfigurationID();
311  Iterator<MotorMount> iterator = config.motorIterator();
312  while (iterator.hasNext()) {
313  MotorMount m = iterator.next();
314  if (m.getIgnitionConfiguration().getDefault().getIgnitionEvent() == IgnitionConfiguration.IgnitionEvent.LAUNCH)
315  launchBurn = MathUtil.max(launchBurn, m.getMotor(id).getBurnTimeEstimate());
316  else
317  otherBurn = otherBurn + m.getMotor(id).getBurnTimeEstimate();
318  }
319  burnoutTimeEstimate = Math.max(launchBurn + otherBurn, 0.1);
320  }

336  @Override
337  protected void process(List<SimulationStatus> chunks) {
338 
339  // Update max. altitude and velocity
340  for (SimulationStatus s : chunks) {
341  simulationMaxAltitude[index] = Math.max(simulationMaxAltitude[index],
342  s.getRocketPosition().z);
343  simulationMaxVelocity[index] = Math.max(simulationMaxVelocity[index],
344  s.getRocketVelocity().length());
345  }
346 
347  // Calculate the progress
348  SimulationStatus status = chunks.get(chunks.size() - 1);
349  simulationStatuses[index] = status;
350 
351  // 1. time = 0 ... burnoutTimeEstimate
352  if (simulationStage == -2 && status.getSimulationTime() < burnoutTimeEstimate) {
353  log.debug("Method 1: t=" + status.getSimulationTime() + " est=" + burnoutTimeEstimate);
354  setSimulationProgress(MathUtil.map(status.getSimulationTime(), 0, burnoutTimeEstimate,
355  0.0, BURNOUT_PROGRESS));
356  updateProgress();
357  return;
358  }
359 
360  if (simulationStage == -2) {
361  simulationStage++;
362  burnoutVelocity = MathUtil.max(status.getRocketVelocity().z, 0.1);
363  log.debug("CHANGING to Method 2, vel=" + burnoutVelocity);
364  }
365 
366  // 2. z-velocity from burnout velocity to zero
367  if (simulationStage == -1 && status.getRocketVelocity().z >= 0) {
368  log.debug("Method 2: vel=" + status.getRocketVelocity().z + " burnout=" + burnoutVelocity);
369  setSimulationProgress(MathUtil.map(status.getRocketVelocity().z, burnoutVelocity, 0,
370  BURNOUT_PROGRESS, APOGEE_PROGRESS));
371  updateProgress();
372  return;
373  }
374 
375  if (simulationStage == -1 && status.getRocketVelocity().z < 0) {
376  simulationStage++;
377  apogeeAltitude = MathUtil.max(status.getRocketPosition().z, 1);
378  log.debug("CHANGING to Method 3, apogee=" + apogeeAltitude);
379  }
380 
381  // 3. z-position from apogee to zero
382  // TODO: MEDIUM: several stages
383  log.debug("Method 3: alt=" + status.getRocketPosition().z + " apogee=" + apogeeAltitude);
384  setSimulationProgress(MathUtil.map(status.getRocketPosition().z,
385  apogeeAltitude, 0, APOGEE_PROGRESS, 1.0));
386  updateProgress();
387  }
