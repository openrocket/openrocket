257  private void updateFilters() {
258  List<RowFilter<TableModel, Object>> filters = new ArrayList<RowFilter<TableModel, Object>>(2);
259  String filterTextRegex = filterText.getText();
260  if (filterTextRegex != null) {
261  try {
262  // The "(?iu)" magic turns on case insensitivity with unicode chars
263  RowFilter<TableModel, Object> regexFilter = RowFilter.regexFilter("(?iu)" + filterTextRegex);
264  filters.add(regexFilter);

266  }

272  filters.add(foreDiameterFilter);
273  }
