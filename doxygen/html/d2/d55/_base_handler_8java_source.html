75  @Override
76  public void closeElement(String element, HashMap<String, String> attributes, String content, WarningSet warnings)
77  throws SAXException {
78  final C component = getComponent();
79  try {
80  if (RocksimCommonConstants.NAME.equals(element)) {
81  component.setName(content);
82  }
83  if (RocksimCommonConstants.KNOWN_MASS.equals(element)) {
84  mass = Math.max(0d, Double.parseDouble(content) / RocksimCommonConstants.ROCKSIM_TO_OPENROCKET_MASS);
85  }
86  if (RocksimCommonConstants.DENSITY.equals(element)) {
87  density = Math.max(0d, Double.parseDouble(content));
88  }
89  if (RocksimCommonConstants.KNOWN_CG.equals(element)) {
90  cg = Math.max(0d, Double.parseDouble(content) / RocksimCommonConstants.ROCKSIM_TO_OPENROCKET_LENGTH);
91  }
92  if (RocksimCommonConstants.USE_KNOWN_CG.equals(element)) { //Rocksim sets UseKnownCG to true to control the override of both cg and mass
93  boolean override = "1".equals(content);
94  setOverride(component, override, mass, cg);
95  }
96  if (RocksimCommonConstants.DENSITY_TYPE.equals(element)) {
97  densityType = RocksimDensityType.fromCode(Integer.parseInt(content));
98  }
99 
100  appearanceBuilder.processElement(element, content, warnings);
101  } catch (NumberFormatException nfe) {
102  warnings.add("Could not convert " + element + " value of " + content + ". It is expected to be a number.");
103  }
104  }
