170  public static void drawExhaust(GL2 gl, FlameSettings fs, Motor motor) {
171  final float s = (float) Math.max(.5, Math.sqrt(motor.getAverageThrustEstimate()) / 4.0)
172  * (float) fs.getExhaustScale();
173  gl.glScalef(s, s, s);
174 
175  gl.glRotated(90, 0, 1, 0);
176  gl.glTranslated(0, 0, 0);
177 
178  gl.glDisable(GLLightingFunc.GL_LIGHTING);
179 
180  if (fs.isSparks() && fs.isFlame()) {
181  sparks(gl, fs);
182  }
183 
184  gl.glEnable(GL.GL_BLEND);
185 
186  gl.glTexEnvi(GL2.GL_TEXTURE_ENV, GL2.GL_TEXTURE_ENV_MODE, GL2.GL_MODULATE);
187  gl.glDepthMask(false);
188 
189  if (fs.isSmoke()) {
190  final float LEN = 10;
191  final float MAX_R = .15f;
192  final int P = 5;
193 
194  final Func radius = new Func() {
195  @Override
196  public float f(double d) {
197  return (float) (Math.atan(d) / (Math.PI / 2.0)) * MAX_R + 0.001f;
198  }
199  };
200 
201  final Func dZ = new Func() {
202  @Override
203  public float f(double z) {
204  return radius.f(z);
205  }
206  };
207 
208  gl.glBlendFunc(GL.GL_SRC_ALPHA, GL.GL_ONE_MINUS_SRC_ALPHA);
209 
210  gl.glActiveTexture(GL.GL_TEXTURE0);
211  smokeT.bind(gl);
212 
213  gl.glActiveTexture(GL.GL_TEXTURE1);
214  smokeN.bind(gl);
215 
216  gl.glUseProgram(shaderprogram);
217 
218  setUniform1i(gl, shaderprogram, "uSmoke", 0);
219  setUniform1i(gl, shaderprogram, "uNormal", 1);
220 
221  trail(gl, radius, dZ, //
222  new Const(0.025f * (fs.getSmokeColor().getAlpha() / 255f)), //
223  LEN, P, fs.getSmokeColor(), s);
224 
225  gl.glUseProgram(0);
226 
227  smokeN.disable(gl);
228  gl.glActiveTexture(GL.GL_TEXTURE0);
229  }
230 
231  if (fs.isFlame()) {
232  gl.glScalef(1, 1, (float) fs.getFlameAspectRatio());
233 
234  gl.glActiveTexture(GL.GL_TEXTURE0);
235  flameT.enable(gl);
236  flameT.bind(gl);
237 
238  final float FLEN = 0.3f;
239  final int FP = 6;
240  final Func fr = new Func() {
241  @Override
242  public float f(double z) {
243  z = z / FLEN;
244  z = 1 - z;
245  return (float) (z * z - z * z * z) * .06f;
246  }
247  };
248 
249  final Func fdZ = new Func() {
250  @Override
251  public float f(double z) {
252  return 0.002f;
253  }
254  };
255 
256  final Func alpha = new Func() {
257  @Override
258  public float f(double z) {
259  return 0.2f * (float) Math.pow((1.0f - (float) z / FLEN), 4);
260  };
261  };
262 
263  gl.glBlendFunc(GL.GL_SRC_ALPHA, GL.GL_ONE);
264  trail(gl, fr, fdZ, alpha, FLEN, FP, fs.getFlameColor(), s);
265 
266  flameT.disable(gl);
267  }
268 
269  gl.glEnable(GLLightingFunc.GL_LIGHTING);
270  gl.glDepthMask(true);
271 
272  }

384  private static void trail(GL2 gl, Func radius, Func dZ, Func alpha, float LEN, int P, Color color, float scale) {
385  float[] c = new float[4];
386  convertColor(color, c);
387 
388  // Figure out if the flame and smoke is point "in" or "out" of the
389  // screen
390  // in order to draw the particles in the right Z order
391  final boolean startAtTop;
392  {
393  final double[] mvmatrix = new double[16];
394  final double[] projmatrix = new double[16];
395  final int[] viewport = new int[4];
396 
397  gl.glGetIntegerv(GL.GL_VIEWPORT, viewport, 0);
398  gl.glGetDoublev(GLMatrixFunc.GL_MODELVIEW_MATRIX, mvmatrix, 0);
399  gl.glGetDoublev(GLMatrixFunc.GL_PROJECTION_MATRIX, projmatrix, 0);
400 
401  final double out[] = new double[4];
402  final double out2[] = new double[4];
403  (new GLU()).gluProject(0, 0, 0, mvmatrix, 0, projmatrix, 0, viewport, 0, out, 0);
404  (new GLU()).gluProject(0, 0, 0.01f, mvmatrix, 0, projmatrix, 0, viewport, 0, out2, 0);
405 
406  startAtTop = out2[2] < out[2];
407  }
408 
409  final float start;
410  final float len;
411  final float mult;
412 
413  if (startAtTop) {
414  start = 0.002f;
415  len = LEN;
416  mult = 1;
417  } else {
418  start = LEN;
419  len = 0.002f;
420  mult = -1;
421  }
422 
423  // Use the same seed every time
424  Random r = new Random(0);
425 
426  // Loop forwards or backwards. Technically the dZ is applied differently
427  // in either direction, but the difference should be vanishingly small.
428  for (float z = start; mult * z < mult * len; z += mult * dZ.f(z)) {
429  gl.glPushMatrix();
430  gl.glTranslatef(0, 0, z);
431 
432  c[3] = alpha.f(z);
433  gl.glColor4fv(c, 0);
434 
435  int[] ii = {0};
436  gl.glGetIntegerv(GL2.GL_CURRENT_PROGRAM, ii, 0);
437  if ( ii[0] == shaderprogram )
438  setUniform1f(gl, shaderprogram, "z", z);
439 
440  for (int i = 0; i < P; i++) {
441  gl.glPushMatrix();
442  float rx = radius.f(z) - ((float) r.nextFloat() * radius.f(z) * 2.0f);
443  float ry = radius.f(z) - ((float) r.nextFloat() * radius.f(z) * 2.0f);
444  float rz = radius.f(z) - ((float) r.nextFloat() * radius.f(z) * 2.0f);
445  gl.glTranslatef(rx, ry, rz);
446 
447  final double[] mvmatrix = new double[16];
448  gl.glGetDoublev(GLMatrixFunc.GL_MODELVIEW_MATRIX, mvmatrix, 0);
449  mvmatrix[0] = mvmatrix[5] = mvmatrix[10] = 1;
450  mvmatrix[1] = mvmatrix[2] = mvmatrix[4] = mvmatrix[6] = mvmatrix[8] = mvmatrix[9] = 0;
451  gl.glLoadMatrixd(mvmatrix, 0);
452 
453  // Add a random rotation to prevent artifacts from texture.
454  gl.glRotatef(r.nextFloat() * 45f, 0, 0, 1);
455 
456  gl.glBegin(GL.GL_TRIANGLE_FAN);
457  float d = radius.f(z) * scale * 2;
458  gl.glTexCoord2f(0, 0);
459  gl.glVertex3f(-d, -d, 0);
460  gl.glTexCoord2f(0, 1);
461  gl.glVertex3f(-d, d, 0);
462  gl.glTexCoord2f(1, 1);
463  gl.glVertex3f(d, d, 0);
464  gl.glTexCoord2f(1, 0);
465  gl.glVertex3f(d, -d, 0);
466  gl.glEnd();
467 
468  gl.glPopMatrix();
469  }
470 
471  gl.glPopMatrix();
472  }
473  }

493  private static void sparks(GL2 gl, FlameSettings fs) {
494  // Use the same seed every time
495  Random r = new Random(0);
496 
497  float[] c = new float[4];
498  float[] c2 = new float[4];
499  convertColor(fs.getFlameColor(), c);
500  for (int i = 0; i < 3; i++) {
501  c[i] = c2[i] = c[i] * .2f + .8f;
502  }
503  c[3] = 1;
504  c2[3] = 1;
505  gl.glBlendFunc(GL.GL_SRC_ALPHA, GL.GL_ONE_MINUS_SRC_ALPHA);
506  gl.glEnable(GL.GL_BLEND);
507  gl.glLineWidth(1 + (float)fs.getSparkWeight() * 1.0f);
508  for (int i = 0; i < 4000 * fs.getSparkConcentration(); i++) {
509  final float z = 0.01f + 2 * (r.nextFloat() * r.nextFloat() * r.nextFloat());
510  final float x = z * (r.nextFloat() - 0.5f);
511  final float y = z * (r.nextFloat() - 0.5f);
512  //gl.glPointSize(1);
513  gl.glBegin(GL.GL_LINES);
514  gl.glColor4fv(c, 0);
515  gl.glVertex3f(x, y, z * 2);
516  gl.glColor4fv(c2, 0);
517  gl.glVertex3f(x * 1.02f, y * 1.02f, z * 2 + 0.01f + ((float)fs.getSparkWeight() / 20));
518  gl.glEnd();
519  }
520  }
