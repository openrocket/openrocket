238  /***
239  * creates evenly spaced values for thumbs.
240  */
241  private int[] createDefaultValues(int min, int max, int num_of_values, boolean useEndPoints) {
242  int[] values = new int[num_of_values];
243  int range = max - min;
244 
245  if (!useEndPoints) {
246  int step = range / (num_of_values + 1);
247  for (int i = 0; i < num_of_values; i++) {
248  values[i] = min + (i + 1) * step;
249  }
250  } else {
251  if (num_of_values < 1) {
252  return new int[0];
253  }
254  values[0] = getMinimum();
255  values[num_of_values - 1] = getMaximum();
256  int[] def = createDefaultValues(getMinimum(), getMaximum(), num_of_values - 2, false);
257  for (int i = 0; i < def.length; i++) {
258  values[i + 1] = def[i];
259  }
260  }
261  return values;
262  }
