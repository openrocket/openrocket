1229  private static boolean open(OpenFileWorker worker, String displayName, Window parent, boolean openRocketConfigDialog) {
1230  // Open the file in a Swing worker thread
1231  log.info("Starting OpenFileWorker");
1232  if (!SwingWorkerDialog.runWorker(parent, "Opening file", "Reading " + displayName + "...", worker)) {
1233  // User cancelled the operation
1234  log.info("User cancelled the OpenFileWorker");
1235  return false;
1236  }
1237 
1238 
1239  // Handle the document
1240  OpenRocketDocument doc = null;
1241  try {
1242 
1243  doc = worker.get();
1244 
1245  } catch (ExecutionException e) {
1246 
1247  Throwable cause = e.getCause();
1248 
1249  if (cause instanceof FileNotFoundException) {
1250

1252  JOptionPane.showMessageDialog(parent,
1253  "File not found: " + displayName,
1254  "Error opening file", JOptionPane.ERROR_MESSAGE);

1258

1263  "Error opening file", JOptionPane.ERROR_MESSAGE);
1264  return false;
1265 
1266  } else {
1267 
1268  throw new BugException("Unknown error when opening file", e);
1269 
1270  }
1271 
1272  } catch (InterruptedException e) {
1273  throw new BugException("EDT was interrupted", e);
1274  }
1275

1277  throw new BugException("Document loader returned null");
1278  }
