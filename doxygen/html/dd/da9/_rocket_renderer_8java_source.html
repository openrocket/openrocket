57  public RocketComponent pick(GLAutoDrawable drawable, Configuration configuration, Point p,
58  Set<RocketComponent> ignore) {
59  final GL2 gl = drawable.getGL().getGL2();
60  gl.glEnable(GL.GL_DEPTH_TEST);
61 
62  // Store a vector of pickable parts.
63  final Vector<RocketComponent> pickParts = new Vector<RocketComponent>();
64 
65  for (RocketComponent c : configuration) {
66  if (ignore != null && ignore.contains(c))
67  continue;
68 
69  // Encode the index of the part as a color
70  // if index is 0x0ABC the color ends up as
71  // 0xA0B0C000 with each nibble in the coresponding
72  // high bits of the RG and B channels.
73  gl.glColor4ub((byte) ((pickParts.size() >> 4) & 0xF0), (byte) ((pickParts.size() << 0) & 0xF0),
74  (byte) ((pickParts.size() << 4) & 0xF0), (byte) 1);
75  pickParts.add(c);
76 
77  if (isDrawnTransparent(c)) {
78  cr.getGeometry(c, Surface.INSIDE).render(gl);
79  } else {
80  cr.getGeometry(c, Surface.ALL).render(gl);
81  }
82  }
83 
84  ByteBuffer bb = ByteBuffer.allocateDirect(4);
85 
86  if (p == null)
87  return null; //Allow pick to be called without a point for debugging
88 
89  gl.glReadPixels(p.x, p.y, 1, 1, GL.GL_RGB, GL.GL_UNSIGNED_BYTE, bb);
90 
91  final int pickColor = bb.getInt();
92  final int pickIndex = ((pickColor >> 20) & 0xF00) | ((pickColor >> 16) & 0x0F0) | ((pickColor >> 12) & 0x00F);
93 
94  log.trace("Picked pixel color is {} index is {}", pickColor, pickIndex);
95 
96  if (pickIndex < 0 || pickIndex > pickParts.size() - 1)
97  return null;
98 
99  return pickParts.get(pickIndex);
100  }
