221  private int calculateNecessaryFileVersion(OpenRocketDocument document, StorageOptions opts) {
222  /*
223  * NOTE: Remember to update the supported versions in DocumentConfig as well!
224  *
225  * File version 1.7 is required for:
226  * - simulation extensions
227  * - saving tube fins.
228  *
229  * File version 1.6 is required for:
230  * - saving files using appearances and textures, flight configurations.
231  *
232  * File version 1.5 is requires for:
233  * - saving designs using ComponentPrests
234  * - recovery device deployment on lower stage separation
235  * - custom expressions
236  *
237  * File version 1.4 is required for:
238  * - saving simulation data
239  * - saving motor data
240  *
241  * File version 1.1 is required for:
242  * - fin tabs
243  * - components attached to tube coupler
244  *
245  * Otherwise use version 1.0.
246  */
247 
249  // Version 1.7 //
251  for (Simulation sim : document.getSimulations()) {
252  if (!sim.getSimulationExtensions().isEmpty()) {
253  return FILE_VERSION_DIVISOR + 7;
254  }
255  }
256 
257  // Search the rocket for any TubeFinSet objects (version 1.7)
258  for (RocketComponent c : document.getRocket()) {
259  if (c instanceof TubeFinSet) {
260  return FILE_VERSION_DIVISOR + 7;
261  }
262  }
263 
264 
266  // Version 1.6 //
268 
269  // Search the rocket for any Appearances or non-motor flight configurations (version 1.6)
270  for (RocketComponent c : document.getRocket()) {
271  if (c.getAppearance() != null) {
272  return FILE_VERSION_DIVISOR + 6;
273  }
274  if (c instanceof FlightConfigurableComponent) {
275  if (c instanceof MotorMount) {
276  MotorMount mmt = (MotorMount) c;
277  if (mmt.getIgnitionConfiguration().size() > 0) {
278  return FILE_VERSION_DIVISOR + 6;
279  }
280  }
281  if (c instanceof RecoveryDevice) {
282  RecoveryDevice recovery = (RecoveryDevice) c;
283  if (recovery.getDeploymentConfiguration().size() > 0) {
284  return FILE_VERSION_DIVISOR + 6;
285  }
286  }
287  if (c instanceof Stage) {
288  Stage stage = (Stage) c;
289  if (stage.getStageSeparationConfiguration().size() > 0) {
290  return FILE_VERSION_DIVISOR + 6;
291  }
292  }
293  }
294  }
295 
297  // Version 1.5 //
299 
300  // Search the rocket for any ComponentPresets (version 1.5)
301  for (RocketComponent c : document.getRocket()) {
302  if (c.getPresetComponent() != null) {
303  return FILE_VERSION_DIVISOR + 5;
304  }
305  }
306 
307  // Search for recovery device deployment type LOWER_STAGE_SEPARATION (version 1.5)
308  for (RocketComponent c : document.getRocket()) {
309  if (c instanceof RecoveryDevice) {
310  if (((RecoveryDevice) c).getDeploymentConfiguration().getDefault().getDeployEvent() == DeployEvent.LOWER_STAGE_SEPARATION) {
311  return FILE_VERSION_DIVISOR + 5;
312  }
313  }
314  }
315 
316  // Check for custom expressions (version 1.5)
317  if (!document.getCustomExpressions().isEmpty()) {
318  return FILE_VERSION_DIVISOR + 5;
319  }
320 
322  // Version 1.4 //
324 
325  // Check if design has simulations defined (version 1.4)
326  if (document.getSimulationCount() > 0) {
327  return FILE_VERSION_DIVISOR + 4;
328  }
329 
330  // Check for motor definitions (version 1.4)
331  for (RocketComponent c : document.getRocket()) {
332  if (!(c instanceof MotorMount))
333  continue;
334 
335  MotorMount mount = (MotorMount) c;
336  for (String id : document.getRocket().getFlightConfigurationIDs()) {
337  if (mount.getMotor(id) != null) {
338  return FILE_VERSION_DIVISOR + 4;
339  }
340  }
341  }
342 
344  // Version 1.3 //
346 
347  // no version 1.3 file type exists
348 
350  // Version 1.2 //
352 
353  // no version 1.2 file type exists
354 
356  // Version 1.1 //
358 
359  // Check for fin tabs or tube coupler children (version 1.1)
360  for (RocketComponent c : document.getRocket()) {
361  // Check for fin tabs
362  if (c instanceof FinSet) {
363  FinSet fin = (FinSet) c;
364  if (!MathUtil.equals(fin.getTabHeight(), 0) &&
365  !MathUtil.equals(fin.getTabLength(), 0)) {
366  return FILE_VERSION_DIVISOR + 1;
367  }
368  }
369 
370  // Check for components attached to tube coupler
371  if (c instanceof TubeCoupler) {
372  if (c.getChildCount() > 0) {
373  return FILE_VERSION_DIVISOR + 1;
374  }
375  }
376  }
377 
379  // Version 1.0 //
381 
382  // Default (version 1.0)
383  return FILE_VERSION_DIVISOR + 0;
384  }
