276  private JComponent findMenu(MenuElement menu, List<String> menuItems) {
277  for (int i = 0; i < menuItems.size() - 1; i++) {
278  String menuItem = menuItems.get(i);
279 
280  MenuElement found = null;
281  for (MenuElement e : menu.getSubElements()) {
282  if (e instanceof JMenu && ((JMenu) e).getText().equals(menuItem)) {
283  found = e;
284  break;
285  }
286  }
287 
288  if (found != null) {
289  menu = found;
290  } else {
291  JMenu m = new JMenu(menuItem);
292  ((JComponent) menu).add(m);
293  menu = m;
294  }
295  }
296  return (JComponent) menu;
