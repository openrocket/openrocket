147  status.getConfiguration().getRocket(),

300  private boolean handleEvents() throws SimulationException {
301  boolean ret = true;
302  FlightEvent event;
303 
304  log.trace("HandleEvents: current branch = " + status.getFlightData().getBranchName());
305  log.trace("EventQueue = " + status.getEventQueue().toString());
306  for (event = nextEvent(); event != null; event = nextEvent()) {
307 
308  // Ignore events for components that are no longer attached to the rocket
309  if (event.getSource() != null && event.getSource().getParent() != null &&
310  !status.getConfiguration().isStageActive(event.getSource().getStageNumber())) {
311  continue;
312  }
313 
314  // Call simulation listeners, allow aborting event handling
315  if (!SimulationListenerHelper.fireHandleFlightEvent(status, event)) {
316  continue;
317  }
318 
319  if (event.getType() != FlightEvent.Type.ALTITUDE) {
320  log.trace("BasicEventSimulationEngine: Handling event " + event);
321  }
322 
323  if (event.getType() == FlightEvent.Type.IGNITION) {
324  MotorMount mount = (MotorMount) event.getSource();
325  MotorId motorId = (MotorId) event.getData();
326  MotorInstance instance = status.getMotorConfiguration().getMotorInstance(motorId);
327  if (!SimulationListenerHelper.fireMotorIgnition(status, motorId, mount, instance)) {
328  continue;
329  }
330  }
331 
332  if (event.getType() == FlightEvent.Type.RECOVERY_DEVICE_DEPLOYMENT) {
333  RecoveryDevice device = (RecoveryDevice) event.getSource();
334  if (!SimulationListenerHelper.fireRecoveryDeviceDeployment(status, device)) {
335  continue;
336  }
337  }
338 
339 
340 
341  // Check for motor ignition events, add ignition events to queue
342  for (MotorId id : status.getMotorConfiguration().getMotorIDs()) {
343  IgnitionConfiguration.IgnitionEvent ignitionEvent = status.getMotorConfiguration().getMotorIgnitionEvent(id);
344  MotorMount mount = status.getMotorConfiguration().getMotorMount(id);
345  RocketComponent component = (RocketComponent) mount;
346 
347  if (ignitionEvent.isActivationEvent(event, component)) {
348  double ignitionDelay = status.getMotorConfiguration().getMotorIgnitionDelay(id);
349  addEvent(new FlightEvent(FlightEvent.Type.IGNITION,
350  status.getSimulationTime() + ignitionDelay,
351  component, id));
352  }
353  }
354 
355 
356  // Check for stage separation event
357  for (int stageNo : status.getConfiguration().getActiveStages()) {
358  if (stageNo == 0)
359  continue;
360 
361  Stage stage = (Stage) status.getConfiguration().getRocket().getChild(stageNo);
362  StageSeparationConfiguration separationConfig = stage.getStageSeparationConfiguration().get(flightConfigurationId);
363  if (separationConfig.getSeparationEvent().isSeparationEvent(event, stage)) {
364  addEvent(new FlightEvent(FlightEvent.Type.STAGE_SEPARATION,
365  event.getTime() + separationConfig.getSeparationDelay(), stage));
366  }
367  }
368 
369 
370  // Check for recovery device deployment, add events to queue
371  Iterator<RocketComponent> rci = status.getConfiguration().iterator();
372  while (rci.hasNext()) {
373  RocketComponent c = rci.next();
374  if (!(c instanceof RecoveryDevice))
375  continue;
376  DeploymentConfiguration deployConfig = ((RecoveryDevice) c).getDeploymentConfiguration().get(flightConfigurationId);
377  if (deployConfig.isActivationEvent(event, c)) {
378  // Delay event by at least 1ms to allow stage separation to occur first
379  addEvent(new FlightEvent(FlightEvent.Type.RECOVERY_DEVICE_DEPLOYMENT,
380  event.getTime() + Math.max(0.001, deployConfig.getDeployDelay()), c));
381  }
382  }
383 
384 
385  // Handle event
386  switch (event.getType()) {
387 
388  case LAUNCH: {
389  status.getFlightData().addEvent(event);
390  break;
391  }
392 
393  case IGNITION: {
394  // Ignite the motor
395  MotorId motorId = (MotorId) event.getData();
396  MotorInstanceConfiguration config = status.getMotorConfiguration();
397  config.setMotorIgnitionTime(motorId, event.getTime());
398  status.setMotorIgnited(true);
399  status.getFlightData().addEvent(event);
400 
401  break;
402  }
403 
404  case LIFTOFF: {
405  // Mark lift-off as occurred
406  status.setLiftoff(true);
407  status.getFlightData().addEvent(event);
408  break;
409  }
410 
411  case LAUNCHROD: {
412  // Mark launch rod as cleared
413  status.setLaunchRodCleared(true);
414  status.getFlightData().addEvent(event);
415  break;
416  }
417 
418  case BURNOUT: {
419  // If motor burnout occurs without lift-off, abort
420  if (!status.isLiftoff()) {
421  throw new SimulationLaunchException(trans.get("BasicEventSimulationEngine.error.earlyMotorBurnout"));
422  }
423  // Add ejection charge event
424  MotorId motorId = (MotorId) event.getData();
425  double delay = status.getMotorConfiguration().getEjectionDelay(motorId);
426  if (delay != Motor.PLUGGED) {
427  addEvent(new FlightEvent(FlightEvent.Type.EJECTION_CHARGE, status.getSimulationTime() + delay,
428  event.getSource(), event.getData()));
429  }
430  status.getFlightData().addEvent(event);
431  break;
432  }
433 
434  case EJECTION_CHARGE: {
435  status.getFlightData().addEvent(event);
436  break;
437  }
438 
439  case STAGE_SEPARATION: {
440  // Record the event.
441  status.getFlightData().addEvent(event);
442 
443  RocketComponent stage = event.getSource();
444  int n = stage.getStageNumber();
445 
446  // Prepare the booster status for simulation.
447  SimulationStatus boosterStatus = new SimulationStatus(status);
448  boosterStatus.setFlightData(new FlightDataBranch(stage.getName(), FlightDataType.TYPE_TIME));
449 
450  stages.add(boosterStatus);
451 
452  // Mark the status as having dropped the booster
453  status.getConfiguration().setToStage(n - 1);
454 
455  // Mark the booster status as only having the booster.
456  boosterStatus.getConfiguration().setOnlyStage(n);
457  break;
458  }
459 
460  case APOGEE:
461  // Mark apogee as reached
462  status.setApogeeReached(true);
463  status.getFlightData().addEvent(event);
464  // This apogee event might be the optimum if recovery has not already happened.
465  if (status.getSimulationConditions().isCalculateExtras() && status.getDeployedRecoveryDevices().size() == 0) {
466  status.getFlightData().setOptimumAltitude(status.getMaxAlt());
467  status.getFlightData().setTimeToOptimumAltitude(status.getMaxAltTime());
468  }
469  break;
470 
471  case RECOVERY_DEVICE_DEPLOYMENT:
472  RocketComponent c = event.getSource();
473  int n = c.getStageNumber();
474  // Ignore event if stage not active
475  if (status.getConfiguration().isStageActive(n)) {
476  // TODO: HIGH: Check stage activeness for other events as well?
477 
478  // Check whether any motor in the active stages is active anymore
479  for (MotorId motorId : status.getMotorConfiguration().getMotorIDs()) {
480  int stage = ((RocketComponent) status.getMotorConfiguration().
481  getMotorMount(motorId)).getStageNumber();
482  if (!status.getConfiguration().isStageActive(stage))
483  continue;
484  if (!status.getMotorConfiguration().getMotorInstance(motorId).isActive())
485  continue;
486  status.getWarnings().add(Warning.RECOVERY_DEPLOYMENT_WHILE_BURNING);
487  }
488 
489  // Check for launch rod
490  if (!status.isLaunchRodCleared()) {
491  status.getWarnings().add(Warning.RECOVERY_LAUNCH_ROD);
492  }
493 
494  // Check current velocity
495  if (status.getRocketVelocity().length() > 20) {
496  status.getWarnings().add(new Warning.HighSpeedDeployment(status.getRocketVelocity().length()));
497  }
498 
499  status.setLiftoff(true);
500  status.getDeployedRecoveryDevices().add((RecoveryDevice) c);
501 
502  // If we haven't already reached apogee, then we need to compute the actual coast time
503  // to determine the optimum altitude.
504  if (status.getSimulationConditions().isCalculateExtras() && !status.isApogeeReached()) {
505  FlightData coastStatus = computeCoastTime();
506  status.getFlightData().setOptimumAltitude(coastStatus.getMaxAltitude());
507  status.getFlightData().setTimeToOptimumAltitude(coastStatus.getTimeToApogee());
508  }
509 
510  this.currentStepper = this.landingStepper;
511  this.status = currentStepper.initialize(status);
512 
513  status.getFlightData().addEvent(event);
514  }
515  break;
516 
517  case GROUND_HIT:
518  status.getFlightData().addEvent(event);
519  break;
520 
521  case SIMULATION_END:
522  ret = false;
523  status.getFlightData().addEvent(event);
524  break;
525 
526  case ALTITUDE:
527  break;
528 
529  case TUMBLE:
530  this.currentStepper = this.tumbleStepper;
531  this.status = currentStepper.initialize(status);
532  status.getFlightData().addEvent(event);
533  break;
534  }
535 
536  }
537 
538 
539  // If no motor has ignited, abort
540  if (!status.isMotorIgnited()) {
541  throw new MotorIgnitionException(trans.get("BasicEventSimulationEngine.error.noIgnition"));
542  }
543 
544  return ret;
545  }
