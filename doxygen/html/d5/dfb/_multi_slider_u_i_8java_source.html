144  protected void calculateThumbLocation() {
145  MultiSlider slider = (MultiSlider) this.slider;
146  int majorTickSpacing = slider.getMajorTickSpacing();
147  int minorTickSpacing = slider.getMinorTickSpacing();
148  int tickSpacing = 0;
149 
150  if (minorTickSpacing > 0) {
151  tickSpacing = minorTickSpacing;
152  } else if (majorTickSpacing > 0) {
153  tickSpacing = majorTickSpacing;
154  }
155  for (int i = 0; i < this.thumbCount; i++) {
156  if (slider.getSnapToTicks()) {
157  int sliderValue = slider.getValueAt(i);
158  int snappedValue = sliderValue;
159  if (tickSpacing != 0) {
160  // If it's not on a tick, change the value
161  if ((sliderValue - slider.getMinimum()) % tickSpacing != 0 ) {
162  float temp = (float)(sliderValue - slider.getMinimum()) / (float)tickSpacing;
163  int whichTick = Math.round(temp);
164  snappedValue = slider.getMinimum() + (whichTick * tickSpacing);
165  }
166 
167  if( snappedValue != sliderValue ) {
168  slider.setValueAt(i, snappedValue);
169  }
170  }
171  }
172 
173  if (slider.getOrientation() == JSlider.HORIZONTAL) {
174  int valuePosition = xPositionForValue(slider.getValueAt(i));
175  this.thumbRects[i].x = valuePosition - (this.thumbRects[i].width / 2);
176  this.thumbRects[i].y = trackRect.y;
177  } else {
178  int valuePosition = yPositionForValue(slider.getValueAt(i));
179  this.thumbRects[i].x = trackRect.x;
180  this.thumbRects[i].y = valuePosition - (this.thumbRects[i].height / 2);
181  }
182  }

392  /*
393  * this is a very lazy way to find the closest. One might want to
394  * implement a much faster algorithm.
395  */
396  private int findClosest(int x, int y, int[] neighbours, int excluded) {
397  int orientation = slider.getOrientation();
398  int rightmin = Integer.MAX_VALUE; // for dxw, dy
399  int leftmin = -Integer.MAX_VALUE; // for dx, dyh
400  int dx = 0;
401  int dxw = 0;
402  int dy = 0;
403  int dyh = 0;
404  neighbours[0] = -1; // left
405  neighbours[1] = -1; // right
406  for (int i = 0; i < MultiSliderUI.this.thumbCount; i++) {
407  if (i == excluded) {
408  continue;
409  }
410  switch (orientation) {
411  case JSlider.VERTICAL:
412  dy = MultiSliderUI.this.thumbRects[i].y - y;
413  dyh = (MultiSliderUI.this.thumbRects[i].y + MultiSliderUI.this.thumbRects[i].height) - y;
414  if (dyh <= 0) {
415  if (dyh > leftmin) { // has to be > and not >=
416  leftmin = dyh;
417  neighbours[0] = i;
418  }
419  }
420  if (dy >= 0) {
421  if (dy <= rightmin) {
422  rightmin = dy;
423  neighbours[1] = i;
424  }
425  }
426  break;
427  case JSlider.HORIZONTAL:
428  dx = MultiSliderUI.this.thumbRects[i].x - x;
429  dxw = (MultiSliderUI.this.thumbRects[i].x + MultiSliderUI.this.thumbRects[i].width) - x;
430  if (dxw <= 0) {
431  if (dxw >= leftmin) {
432  leftmin = dxw;
433  neighbours[0] = i;
434  }
435  }
436  if (dx >= 0) {
437  if (dx < rightmin) { // has to be < and not <=
438  rightmin = dx;
439  neighbours[1] = i;
440  }
441  }
442  break;
443  }
444  }
445  //System.out.println("neighbours = " + neighbours[0] + ", " + neighbours[1]);
446  int closest = (Math.abs(leftmin) <= Math.abs(rightmin)) ? neighbours[0] : neighbours[1];
447  return (closest == -1) ? 0 : closest;
448  }
