487  @Override
488  public void stateChanged(EventObject e) {
489  AerodynamicForces forces;
490  WarningSet set = new WarningSet();
491  conditions.setAOA(aoa.getValue());
492  conditions.setTheta(theta.getValue());
493  conditions.setMach(mach.getValue());
494  conditions.setRollRate(roll.getValue());
495  conditions.setReference(configuration);
496 
497  if (worstToggle.isSelected()) {
498  aerodynamicCalculator.getWorstCP(configuration, conditions, null);
499  if (!MathUtil.equals(conditions.getTheta(), theta.getValue())) {
500  fakeChange = true;
501  theta.setValue(conditions.getTheta()); // Fires a stateChanged event
502  fakeChange = false;
503  return;
504  }
505  }
506 
507  Map<RocketComponent, AerodynamicForces> aeroData =
508  aerodynamicCalculator.getForceAnalysis(configuration, conditions, set);
509  Map<RocketComponent, Coordinate> massData =
510  massCalculator.getCGAnalysis(configuration, MassCalcType.LAUNCH_MASS);
511 
512 
513  cpData.clear();
514  cgData.clear();
515  dragData.clear();
516  rollData.clear();
517  for (RocketComponent c : configuration) {
518  forces = aeroData.get(c);
519  Coordinate cg = massData.get(c);
520 
521  if (forces == null)
522  continue;
523  if (forces.getCP() != null) {
524  cpData.add(forces);
525  cgData.add(cg);
526  }
527  if (!Double.isNaN(forces.getCD())) {
528  dragData.add(forces);
529  }
530  if (c instanceof FinSet) {
531  rollData.add(forces);
532  }
533  }
534  forces = aeroData.get(configuration.getRocket());
535  if (forces != null) {
536  cpData.add(forces);
537  cgData.add(massData.get(configuration.getRocket()));
538  dragData.add(forces);
539  rollData.add(forces);
540  totalCD = forces.getCD();
541  } else {
542  totalCD = 0;
543  }
544 
545  // Set warnings
546  if (set.isEmpty()) {
547  warningList.setListData(new String[] {
548  trans.get("componentanalysisdlg.noWarnings")
549  });
550  } else {
551  warningList.setListData(new Vector<Warning>(set));
552  }
553 
554  cpTableModel.fireTableDataChanged();
