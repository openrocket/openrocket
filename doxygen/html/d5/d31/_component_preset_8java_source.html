330  void computeDigest() {
331 
332  try {
333  ByteArrayOutputStream bos = new ByteArrayOutputStream();
334  DataOutputStream os = new DataOutputStream(bos);
335 
336  List<TypedKey<?>> keys = new ArrayList<TypedKey<?>>(properties.keySet());
337 
338  Collections.sort(keys, new Comparator<TypedKey<?>>() {
339  @Override
340  public int compare(TypedKey<?> a, TypedKey<?> b) {
341  return a.getName().compareTo(b.getName());
342  }
343  });
344 
345  for (TypedKey<?> key : keys) {
346 
347  Object value = properties.get(key);
348 
349  os.writeBytes(key.getName());
350 
351  if (key.getType() == Double.class) {
352  Double d = (Double) value;
353  os.writeDouble(d);
354  } else if (key.getType() == String.class) {
355  String s = (String) value;
356  os.writeBytes(s);
357  } else if (key.getType() == Manufacturer.class) {
358  String s = ((Manufacturer) value).getSimpleName();
359  os.writeBytes(s);
360  } else if (key.getType() == Finish.class) {
361  String s = ((Finish) value).name();
362  os.writeBytes(s);
363  } else if (key.getType() == Type.class) {
364  String s = ((Type) value).name();
365  os.writeBytes(s);
366  } else if (key.getType() == Boolean.class) {
367  Boolean b = (Boolean) value;
368  os.writeBoolean(b);
369  } else if (key.getType() == Material.class) {
370  double d = ((Material) value).getDensity();
371  os.writeDouble(d);
372  } else if (key.getType() == Shape.class) {
373  // this is ugly to use the ordinal but what else?
374  int i = ((Shape) value).ordinal();
375  os.writeInt(i);
376  }
377 
378  }
379 
380  MessageDigest md5 = MessageDigest.getInstance("MD5");
381  digest = TextUtil.hexString(md5.digest(bos.toByteArray()));
382  } catch (Exception e) {
383  e.printStackTrace();
384  throw new BugException(e);
385  }
